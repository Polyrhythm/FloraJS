  2 /**
  3  * Creates a new Attractor object.
  4  *
  5  * @constructor
  6  * @extends Agent
  7  *
  8  * @param {Object} [opt_options] Options.
  9  * @param {number} [opt_options.G = -1] Universal Gravitational Constant.
 10  * @param {number} [opt_options.mass = 100] Mass. Increase for a greater gravitational effect.
 11  * @param {boolean} [opt_options.isStatic = true] If true, object will not move.
 12  * @param {number} [opt_options.width = 10] Width.
 13  * @param {number} [opt_options.height = 10] Height.
 14  * @param {number} [opt_options.opacity = 0.75] The particle's opacity.
 15  */
 16 function Attractor(opt_options) {
 17 
 18   'use strict';
 19 
 20   var options = opt_options || {};
 21 
 22   exports.Agent.call(this, options);
 23 
 24   this.G = options.G === 0 ? 0 : options.G || 1;
 25   this.mass = options.mass === 0 ? 0 : options.mass || 100;
 26   this.isStatic = options.isStatic === false ? false : options.isStatic || true;
 27   this.width = options.width === 0 ? 0 : options.width || 50;
 28   this.height = options.height === 0 ? 0 : options.height || 50;
 29   this.opacity = options.opacity === 0 ? 0 : options.opacity || 0.75;
 30 }
 31 exports.Utils.extend(Attractor, exports.Agent);
 32 
 33 /**
 34  * Define a name property.
 35  */
 36 Attractor.prototype.name = 'attractor';
 37 
 38 exports.Attractor = Attractor;
