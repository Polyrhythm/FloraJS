  2 /**
  3  * Creates a new Oscillator.
  4  * Oscillators simulate wave patterns and move according to
  5  * amplitude and angular velocity.
  6  *
  7  * @constructor
  8  * @extends Agent
  9  *
 10  * @param {Object} [opt_options] Oscillator options.
 11  * @param {Object} [opt_options.initialLocation = The center of the world] The object's initial location.
 12  * @param {Object} [opt_options.lastLocation = {x: 0, y: 0}] The object's last location. Used to calculate
 13  *    angle if pointToDirection = true.
 14  * @param {number} [opt_options.width = 10] Width
 15  * @param {number} [opt_options.height = 10] Height
 16  * @param {Object} [opt_options.amplitude = {x: 4, y: 0}] Sets amplitude, the distance from the object's
 17  *    initial location (center of the motion) to either extreme.
 18  * @param {Object} [opt_options.acceleration = {x: 0, y: 0}] The object's acceleration. Oscillators have a
 19  *    constant acceleration.
 20  * @param {boolean} [opt_options.isStatic = false] If true, object will not move.
 21  * @param {boolean} [opt_options.isPerlin = true] If set to true, object will use Perlin Noise to calculate its location.
 22  * @param {number} [opt_options.perlinSpeed = 0.005] If isPerlin = true, perlinSpeed determines how fast the object location moves through the noise space.
 23  * @param {number} [opt_options.perlinTime = 0] Sets the Perlin Noise time.
 24  * @param {number} [opt_options.perlinAccelLow = -0.075] The lower bound of acceleration when isPerlin = true.
 25  * @param {number} [opt_options.perlinAccelHigh = 0.075] The upper bound of acceleration when isPerlin = true.
 26  * @param {number} [opt_options.offsetX = Math.random() * 10000] The x offset in the Perlin Noise space.
 27  * @param {number} [opt_options.offsetY = Math.random() * 10000] The y offset in the Perlin Noise space.
 28  */
 29 function Oscillator(opt_options) {
 30 
 31   'use strict';
 32 
 33   var options = opt_options || {};
 34 
 35   exports.Agent.call(this, options);
 36 
 37   this.initialLocation = options.initialLocation ||
 38       new exports.Vector(this.world.width/2, this.world.height/2);
 39   this.lastLocation = new exports.Vector(0, 0);
 40   this.width = options.width === 0 ? 0 : options.width || 10;
 41   this.height = options.height === 0 ? 0 : options.height || 10;
 42   this.amplitude = options.amplitude || new exports.Vector(this.world.width/2 - this.width, this.world.height/2 - this.height);
 43   this.acceleration = options.acceleration || new exports.Vector(0.01, 0);
 44   this.aVelocity = new exports.Vector(0, 0);
 45   this.isStatic = !!options.isStatic;
 46 
 47   this.isPerlin = !!options.isPerlin;
 48   this.perlinSpeed = options.perlinSpeed || 0.005;
 49   this.perlinTime = options.perlinTime || 0;
 50   this.perlinAccelLow = options.perlinAccelLow || -2;
 51   this.perlinAccelHigh = options.perlinAccelHigh || 2;
 52   this.perlinOffsetX = options.perlinOffsetX || Math.random() * 10000;
 53   this.perlinOffsetY = options.perlinOffsetY || Math.random() * 10000;
 54 }
 55 exports.Utils.extend(Oscillator, exports.Agent);
 56 
 57 Oscillator.prototype.name = 'Oscillator';
 58 
 59 /**
 60  * Updates the oscillator's properties.
 61  */
 62 Oscillator.prototype.step = function () {
 63 
 64   'use strict';
 65 
 66   var world = this.world, velDiff;
 67 
 68   if (this.beforeStep) {
 69     this.beforeStep.apply(this);
 70   }
 71 
 72   if (!this.isStatic && !this.isPressed) {
 73 
 74     if (this.isPerlin) {
 75       this.perlinTime += this.perlinSpeed;
 76       this.aVelocity.x =  exports.Utils.map(exports.SimplexNoise.noise(this.perlinTime + this.perlinOffsetX, 0, 0.1), -1, 1, this.perlinAccelLow, this.perlinAccelHigh);
 77       this.aVelocity.y =  exports.Utils.map(exports.SimplexNoise.noise(0, this.perlinTime + this.perlinOffsetY, 0.1), -1, 1, this.perlinAccelLow, this.perlinAccelHigh);
 78     } else {
 79       this.aVelocity.add(this.acceleration); // add acceleration
 80     }
 81 
 82     this.location.x = this.initialLocation.x + Math.sin(this.aVelocity.x) * this.amplitude.x;
 83     this.location.y = this.initialLocation.y + Math.sin(this.aVelocity.y) * this.amplitude.y;
 84 
 85     if (this.pointToDirection) { // object rotates toward direction
 86         velDiff = exports.Vector.VectorSub(this.location, this.lastLocation);
 87         this.angle = exports.Utils.radiansToDegrees(Math.atan2(velDiff.y, velDiff.x));
 88     }
 89 
 90     if (this.controlCamera) {
 91       this.checkCameraEdges();
 92     }
 93 
 94     if (this.checkEdges || this.wrapEdges) {
 95       this.checkWorldEdges(world);
 96     }
 97 
 98     if (this.lifespan > 0) {
 99       this.lifespan -= 1;
100     }
101 
102     if (this.afterStep) {
103       this.afterStep.apply(this);
104     }
105 
106     if (this.pointToDirection) {
107       this.lastLocation.x = this.location.x;
108       this.lastLocation.y = this.location.y;
109     }
110   }
111 };
112 exports.Oscillator = Oscillator;
