  9 /**
 10  * Use to extend the properties and methods of a superClass
 11  * onto a subClass.
 12  */
 13 Utils.extend = function(subClass, superClass) {
 14 
 15   'use strict';
 16 
 17   function F() {}
 18   F.prototype = superClass.prototype;
 19   subClass.prototype = new F;
 20   subClass.prototype.constructor = subClass;
 21 };

 23 /**
 24  * Re-maps a number from one range to another.
 25  *
 26  * @param {number} value The value to be converted.
 27  * @param {number} min1 Lower bound of the value's current range.
 28  * @param {number} max1 Upper bound of the value's current range.
 29  * @param {number} min2 Lower bound of the value's target range.
 30  * @param {number} max2 Upper bound of the value's target range.
 31  * @returns {number} A number.
 32  */
 33 Utils.map = function(value, min1, max1, min2, max2) { // returns a new value relative to a new range
 34 
 35   'use strict';
 36 
 37   var unitratio = (value - min1) / (max1 - min1);
 38   return (unitratio * (max2 - min2)) + min2;
 39 };

 41 /**
 42  * Generates a psuedo-random number within a range.
 43  *
 44  * @param {number} low The low end of the range.
 45  * @param {number} high The high end of the range.
 46  * @param {boolean} [flt] Set to true to return a float.
 47  * @returns {number} A number.
 48  */
 49 Utils.getRandomNumber = function(low, high, flt) {
 50 
 51   'use strict';
 52 
 53   if (flt) {
 54     return Math.random()*(high-(low-1)) + low;
 55   }
 56   return Math.floor(Math.random()*(high-(low-1))) + low;
 57 };

 59 /**
 60  * Converts degrees to radians.
 61  *
 62  * @param {number} degrees The degrees value to be converted.
 63  * @returns {number} A number in radians.
 64  */
 65 Utils.degreesToRadians = function(degrees) {
 66 
 67   'use strict';
 68 
 69   if (typeof degrees !== 'undefined') {
 70     return 2 * Math.PI * (degrees/360);
 71   } else {
 72     if (typeof console !== 'undefined') {
 73       console.log('Error: Utils.degreesToRadians is missing degrees param.');
 74     }
 75     return false;
 76   }
 77 };

 99 /**
100  * Constrain a value within a range.
101  *
102  * @param {number} val The value to constrain.
103  * @param {number} low The lower bound of the range.
104  * @param {number} high The upper bound of the range.
105  * @returns {number} A number.
106  */
107 Utils.constrain = function(val, low, high) {
108 
109   'use strict';
110 
111   if (val > high) {
112     return high;
113   } else if (val < low) {
114     return low;
115   }
116   return val;
117 };

135 /**
136  * Add an event listener to a DOM element.
137  *
138  * @param {Object} target The element to receive the event listener.
139  * @param {string} eventType The event type.
140  * @param {function} The function to run when the event is triggered.
141  */
142 Utils.addEvent = function(target, eventType, handler) {
143 
144   'use strict';
145 
146   if (target.addEventListener) { // W3C
147     this.addEventHandler = function(target, eventType, handler) {
148       target.addEventListener(eventType, handler, false);
149     };
150   } else if (target.attachEvent) { // IE
151     this.addEventHandler = function(target, eventType, handler) {
152       target.attachEvent("on" + eventType, handler);
153     };
154   }
155   this.addEventHandler(target, eventType, handler);
156 };

158 /**
159  * Logs a message to the browser console.
160  *
161  * @param {string} msg The message to log.
162  */
163 Utils.log = function(msg) {
164 
165   'use strict';
166 
167   if (typeof console !== 'undefined' && typeof console.log !== 'undefined') {
168     this.log = function(msg) {
169       console.log(msg); // output error to console
170     };
171     this.log.call(this, msg);
172   } else {
173    this.log = function () {}; // noop
174   }
175 };

177 /**
178  * @returns {Object} The current window width and height.
179  * @example getWindowDim() returns {width: 1024, height: 768}
180  */
181 Utils.getWindowSize = function() {
182 
183   'use strict';
184 
185   var d = {
186     'width' : false,
187     'height' : false
188   };
189   if (typeof(window.innerWidth) !== "undefined") {
190     d.width = window.innerWidth;
191   } else if (typeof(document.documentElement) !== "undefined" &&
192       typeof(document.documentElement.clientWidth) !== "undefined") {
193     d.width = document.documentElement.clientWidth;
194   } else if (typeof(document.body) !== "undefined") {
195     d.width = document.body.clientWidth;
196   }
197   if (typeof(window.innerHeight) !== "undefined") {
198     d.height = window.innerHeight;
199   } else if (typeof(document.documentElement) !== "undefined" &&
200       typeof(document.documentElement.clientHeight) !== "undefined") {
201     d.height = document.documentElement.clientHeight;
202   } else if (typeof(document.body) !== "undefined") {
203     d.height = document.body.clientHeight;
204   }
205   return d;
206 };

208 /**
209  * Concatenates several properties into a single list
210  * representing the style properties of an object.
211  *
212  * @param [Object] props A map of properties.
213  * @returns {string} A list of style properties.
214  */
215 Utils.getCSSText = function(props) {
216 
217   'use strict';
218 
219   var positionStr = '';
220 
221   if (!props.color) {
222     props.color = [];
223     props.background = null;
224   } else {
225     props.background = props.cm + '(' + props.color[0] + ', ' + props.color[1] + ', ' + props.color[2] + ')';
226   }
227 
228   if (!props.borderColor) {
229     props.borderColor = [];
230     props.borderColorStr = '';
231   } else {
232     props.borderColorStr = props.cm + '(' + props.borderColor[0] + ', ' + props.borderColor[1] + ', ' + props.borderColor[2] + ')';
233   }
234 
235   if (Modernizr.csstransforms3d) {
236     positionStr = [
237       '-webkit-transform: translateX(' + props.x + 'px) translateY(' + props.y + 'px) translateZ(0) rotate(' + props.a + 'deg) scaleX(' + props.s + ') scaleY(' + props.s + ')',
238       '-moz-transform: translateX(' + props.x + 'px) translateY(' + props.y + 'px) translateZ(0) rotate(' + props.a + 'deg) scaleX(' + props.s + ') scaleY(' + props.s + ')',
239       '-o-transform: translateX(' + props.x + 'px) translateY(' + props.y + 'px) translateZ(0) rotate(' + props.a + 'deg) scaleX(' + props.s + ') scaleY(' + props.s + ')',
240       '-ms-transform: translateX(' + props.x + 'px) translateY(' + props.y + 'px) translateZ(0) rotate(' + props.a + 'deg) scaleX(' + props.s + ') scaleY(' + props.s + ')'
241     ].join(';');
242   } else if (Modernizr.csstransforms) {
243     positionStr = [
244       '-webkit-transform: translateX(' + props.x + 'px) translateY(' + props.y + 'px) rotate(' + props.a + 'deg) scaleX(' + props.s + ') scaleY(' + props.s + ')',
245       '-moz-transform: translateX(' + props.x + 'px) translateY(' + props.y + 'px) rotate(' + props.a + 'deg) scaleX(' + props.s + ') scaleY(' + props.s + ')',
246       '-o-transform: translateX(' + props.x + 'px) translateY(' + props.y + 'px) rotate(' + props.a + 'deg) scaleX(' + props.s + ') scaleY(' + props.s + ')',
247       '-ms-transform: translateX(' + props.x + 'px) translateY(' + props.y + 'px) rotate(' + props.a + 'deg) scaleX(' + props.s + ') scaleY(' + props.s + ')'
248     ].join(';');
249   } else {
250     positionStr = [
251       'position: absolute',
252       'left: ' + props.x + 'px',
253       'top: ' + props.y + 'px'
254     ].join(';');
255   }
256 
257   return [
258     positionStr,
259     'opacity: ' + props.o,
260     'width: ' + props.w + 'px',
261     'height: ' + props.h + 'px',
262     'background: ' + props.background,
263     'z-index: ' + props.z,
264     'border-width: ' + props.borderWidth + 'px',
265     'border-style: ' + props.borderStyle,
266     'border-color: ' + props.borderColorStr,
267     'border-radius: ' + props.borderRadius,
268     'box-shadow: ' + props.boxShadow
269   ].join(';');
270 };
