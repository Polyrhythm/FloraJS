  2 /**
  3  * Creates a new FeatureDetector.
  4  *
  5  * @constructor
  6  */
  7 function FeatureDetector(options) {
  8   'use strict';
  9 }
 10 
 11 FeatureDetector.prototype.name = 'FeatureDetector';
 12 
 13 /**
 14  * Checks if the class has a method to detect the passed feature.
 15  * If so, it calls the method.
 16  *
 17  * @param {string} feature The feature to check.
 18  * @returns True if the feature is supported, false if not.
 19  */
 20 FeatureDetector.prototype.detect = function(feature) {
 21 
 22   'use strict';
 23 
 24   if (!this[feature]) {
 25     return false;
 26   }
 27 
 28   return this[feature].call(this);
 29 };

 31 /**
 32  * Checks if CSS Transforms are supported.
 33  *
 34  * @returns True if the feature is supported, false if not.
 35  */
 36 FeatureDetector.prototype.csstransforms = function() {
 37 
 38   'use strict';
 39 
 40   var transforms = [
 41     '-webkit-transform: translateX(1px) translateY(1px)',
 42     '-moz-transform: translateX(1px) translateY(1px)',
 43     '-o-transform: translateX(1px) translateY(1px)',
 44     '-ms-transform: translateX(1px) translateY(1px)'
 45   ].join(';');
 46 
 47   var docFrag = document.createDocumentFragment();
 48   var div = document.createElement('div');
 49   docFrag.appendChild(div);
 50   div.style.cssText = transforms;
 51 
 52   var styles = [
 53     div.style.transform,
 54     div.style.webkitTransform,
 55     div.style.MozTransform,
 56     div.style.OTransform,
 57     div.style.msTransform
 58   ];
 59   var check = false;
 60 
 61   for (var i = 0; i < styles.length; i += 1) {
 62     if (styles[i]) {
 63       check = true;
 64       break;
 65     }
 66   }
 67 
 68   return check;
 69 };
