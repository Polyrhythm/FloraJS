  3  * Creates a new Agent and appends it to Flora.elements.
  4  *
  5  * @constructor
  6  * @extends Obj
  7  *
  8  * @param {Object} [opt_options] Agent options.
  9  * @param {string} [opt_options.id = "m-" + Agent._idCount] An id. If an id is not provided, one is created.
 10  * @param {Object|function} [opt_options.view] HTML representing the Agent instance.
 11  * @param {string} [opt_options.className = 'agent'] The corresponding DOM element's class name.
 12  * @param {number} [opt_options.mass = 10] Mass
 13  * @param {number} [opt_options.maxSpeed = 10] Maximum speed
 14  * @param {number} [opt_options.minSpeed = 0] Minimum speed
 15  * @param {number} [opt_options.motorSpeed = 2] Motor speed
 16  * @param {number} [opt_options.scale = 1] Scale
 17  * @param {number} [opt_options.angle = 0] Angle
 18  * @param {number} [opt_options.opacity = 0.85] Opacity
 19  * @param {number} [opt_options.lifespan = -1] Life span. Set to -1 to live forever.
 20  * @param {number} [opt_options.width = 20] Width
 21  * @param {number} [opt_options.height = 20] Height
 22  * @param {number} [opt_options.offsetDistance = 30] The distance from the center of the agent's parent.
 23  * @param {number} [opt_options.offsetAngle = 30] The angle of rotation around the parent carrying the agent.
 24  * @param {string} [opt_options.colorMode = 'rgb'] Color mode. Valid options are 'rgb'. 'hex' and 'hsl' coming soon.
 25  * @param {Array} [opt_options.color = null] The object's color expressed as an rbg or hsl value. ex: [255, 100, 0]
 26  * @param {number} [opt_options.zIndex = 1] z-index
 27  * @param {boolean} [opt_options.pointToDirection = true] If true, object will point in the direction it's moving.

 29  * @param {boolean} [opt_options.isStatic = false] If true, object will not move.

 33  * @param {boolean} [opt_options.avoidEdges = false] Set to true to calculate a steering force away from the
 34  * world's bounds.
 35  * @param {number} [opt_options.avoidEdgesStrength = 200] Sets the strength of the steering force when avoidEdges = true.
 36  * @param {number} [opt_options.bounciness = 0.75] Set the strength of the rebound when an object is outside the

 39  * @param {number} [opt_options.turningRadius = 60] Used to calculate steering force with key control.
 40  * @param {number} [opt_options.thrust = 5] Used to apply forward motion with key control.
 41  * @param {boolean} [opt_options.flocking = false] Set to true to apply flocking forces to this object.
 42  * @param {number} [opt_options.desiredSeparation = Twice the object's default width] Sets the desired separation from other objects when flocking = true.
 43  * @param {number} [opt_options.separateStrength = 1] The strength of the force to apply to separating when flocking = true.
 44  * @param {number} [opt_options.alignStrength = 1] The strength of the force to apply to aligning when flocking = true.
 45  * @param {number} [opt_options.cohesionStrength = 1] The strength of the force to apply to cohesion when flocking = true.
 46  * @param {array} [opt_options.sensors = []] A list of sensors attached to this object.
 47  * @param {Object} [opt_options.flowField = null] If a flow field is set, object will use it to apply a force.
 48  * @param {function} [opt_options.beforeStep = ''] A function to run before the step() function.
 49  * @param {function} [opt_options.afterStep = ''] A function to run after the step() function.
 50  * @param {Object} [opt_options.acceleration = {x: 0, y: 0}] The object's initial acceleration.
 51  * @param {Object} [opt_options.velocity = {x: 0, y: 0}] The object's initial velocity.
 52  * @param {Object} [opt_options.location = The center of the world] The object's initial location.

196 /**
197  * Called every frame, step() updates the instance's properties.
198  */
199 Agent.prototype.step = function() {
200 
201   'use strict';
202 
203   var i, max, dir, friction, force, nose, r, theta, x, y, sensor, className, sensorActivated,
204     world = this.world, elements = exports.elementList.all();
205 
206   //
207 
208   if (this.beforeStep) {
209     this.beforeStep.apply(this);
210   }
211 
212   //
213 
214   if (!this.isStatic && !this.isPressed) {
215 
216     // APPLY FORCES -- start
217 
218     if (exports.liquids.length > 0) { // liquid
219       for (i = 0, max = exports.liquids.length; i < max; i += 1) {
220         if (this.id !== exports.liquids[i].id && this.isInside(exports.liquids[i])) {
221           force = this.drag(exports.liquids[i]);
222           this.applyForce(force);
223           className = exports.liquids[i].className + ' activated';
224           exports.liquids[i].el.className = className;
225         }
226       }
227     }
228 
229     if (exports.repellers.length > 0) { // repeller
230       for (i = 0, max = exports.repellers.length; i < max; i += 1) {
231         if (this.id !== exports.repellers[i].id) {
232           force = this.attract(exports.repellers[i]);
233           this.applyForce(force);
234         }
235       }
236     }
237 
238     if (exports.attractors.length > 0) { // attractor
239       for (i = 0, max = exports.attractors.length; i < max; i += 1) {
240         if (this.id !== exports.attractors[i].id) {
241           force = this.attract(exports.attractors[i]);
242           this.applyForce(force);
243         }
244       }
245     }
246 
247     if (this.sensors.length > 0) { // Sensors
248       for (i = 0, max = this.sensors.length; i < max; i += 1) {
249 
250         sensor = this.sensors[i];
251 
252         r = sensor.offsetDistance; // use angle to calculate x, y
253         theta = exports.Utils.degreesToRadians(this.angle + sensor.offsetAngle);
254         x = r * Math.cos(theta);
255         y = r * Math.sin(theta);
256 
257         sensor.location.x = this.location.x;
258         sensor.location.y = this.location.y;
259         sensor.location.add(new exports.Vector(x, y)); // position the sensor
260 
261         if (sensor.activated) {
262           this.applyForce(sensor.getActivationForce({
263             agent: this
264           }));
265           sensorActivated = true;
266         }
267 
268       }
269     }
270 
271     /**
272      * If no sensor were activated and this.motorSpeed != 0,
273      * apply a force in the direction of the current velocity.
274      */
275     if (!sensorActivated && this.motorSpeed) {
276       dir = exports.Utils.clone(this.velocity);
277       dir.normalize();
278       if (this.velocity.mag() > this.motorSpeed) { // decelerate to defaultSpeed
279         dir.mult(-this.motorSpeed);
280       } else {
281         dir.mult(this.motorSpeed);
282       }
283       this.applyForce(dir); // constantly applies a force
284     }
285 
286     if (world.c) { // friction
287       friction = exports.Utils.clone(this.velocity);
288       friction.mult(-1);
289       friction.normalize();
290       friction.mult(world.c);
291       this.applyForce(friction);
292     }
293 
294     this.applyForce(world.wind); // wind
295 
296     this.applyForce(world.gravity); // gravity
297 
298     if (this.followMouse) { // follow mouse
299       var t = {
300         location: new exports.Vector(exports.mouse.loc.x, exports.mouse.loc.y)
301       };
302       this.applyForce(this.seek(t));
303     }
304 
305     if (this.target) { // follow target
306       this.applyForce(this.seek(this.target));
307     }
308 
309     if (this.flowField) { // follow flow field
310       var res = this.flowField.resolution,
311         col = Math.floor(this.location.x/res),
312         row = Math.floor(this.location.y/res),
313         loc, target;
314 
315       if (this.flowField.field[col]) {
316         loc = this.flowField.field[col][row];
317         if (loc) { // !! sometimes loc is not available for edge cases; need to fix
318           target = {
319             location: new exports.Vector(loc.x, loc.y)
320           };
321         } else {
322           target = {
323             location: new exports.Vector(this.location.x, this.location.y)
324           };
325         }
326         this.applyForce(this.follow(target));
327       }
328 
329     }
330 
331     if (this.flocking) {
332       this.flock(elements);
333     }
334 
335     // end -- APPLY FORCES
336 
337     this.velocity.add(this.acceleration); // add acceleration
338 
339     if (this.maxSpeed) {
340       this.velocity.limit(this.maxSpeed); // check if velocity > maxSpeed
341     }
342 
343     if (this.minSpeed) {
344       this.velocity.limitLow(this.minSpeed); // check if velocity < minSpeed
345     }
346 
347     this.location.add(this.velocity); // add velocity
348 
349     if (this.pointToDirection) { // object rotates toward direction
350       if (this.velocity.mag() > 0.1) {
351         this.angle = exports.Utils.radiansToDegrees(Math.atan2(this.velocity.y, this.velocity.x));
352       }
353     }
354 
355     if (this.controlCamera) { // check camera after velocity calculation
356       this.checkCameraEdges();
357     }
358 
359     if (this.checkEdges || this.wrapEdges) {
360       this.checkWorldEdges(world);
361     }
362 
363 
364     if (this.parent) { // parenting
365 
366         if (this.offsetDistance) {
367 
368           r = this.offsetDistance; // use angle to calculate x, y
369           theta = exports.Utils.degreesToRadians(this.parent.angle + this.offsetAngle);
370           x = r * Math.cos(theta);
371           y = r * Math.sin(theta);
372 
373           this.location.x = this.parent.location.x;
374           this.location.y = this.parent.location.y;
375           this.location.add(new exports.Vector(x, y)); // position the sensor
376 
377         } else {
378           this.location = this.parent.location;
379         }
380 
381     }
382 
383     //
384 
385     if (this.afterStep) {
386       this.afterStep.apply(this);
387     }
388 
389     this.acceleration.mult(0); // reset acceleration
390 
391     if (this.lifespan > 0) {
392       this.lifespan -= 1;
393     }
394   }
395 };

397 /**
398  * Applies a force to this object's acceleration.
399  *
400  * @param {Object} force The force to be applied (expressed as a vector).
401  */
402 Agent.prototype.applyForce = function(force) {
403 
404   'use strict';
405 
406   // F = M * A
407   var f = force.clone(); // make a copy of the force so the original force vector is not altered by dividing by mass; could also use static method
408 
409   f.div(this.mass);
410   this.acceleration.add(f);
411 };

413 /**
414  * Calculates a steering force to apply to an object seeking another object.
415  *
416  * @param {Object} target The object to seek.
417  * @param {boolean} arrive Set to true to for this object to arrive and stop at the target.
418  * @returns {Object} The force to apply.
419  */
420 Agent.prototype.seek = function(target, arrive) {
421 
422   'use strict';
423 
424   var world = this.world,
425     desiredVelocity = exports.Vector.VectorSub(target.location, this.location),
426     distanceToTarget = desiredVelocity.mag();
427 
428   desiredVelocity.normalize();
429 
430   if (distanceToTarget < world.width/2) {
431     var m = exports.Utils.map(distanceToTarget, 0, world.width/2, 0, this.maxSpeed);
432     desiredVelocity.mult(m);
433   } else {
434     desiredVelocity.mult(this.maxSpeed);
435   }
436 
437   var steer = exports.Vector.VectorSub(desiredVelocity, this.velocity);
438   steer.limit(this.maxSteeringForce);
439   return steer;
440 };

448 Agent.prototype.follow = function(target) {
449 
450   'use strict';

452   var desiredVelocity = target.location;
453 
454   desiredVelocity.mult(this.maxSpeed);
455 
456   var steer = exports.Vector.VectorSub(desiredVelocity, this.velocity);
457   steer.limit(this.maxSteeringForce);
458   return steer;

461 /**
462  * Bundles flocking behaviors (separate, align, cohesion) into one call.
463  */
464 Agent.prototype.flock = function(elements) {
465 
466   'use strict';
467 
468   this.applyForce(this.separate(elements).mult(this.separateStrength));
469   this.applyForce(this.align(elements).mult(this.alignStrength));
470   this.applyForce(this.cohesion(elements).mult(this.cohesionStrength));
471 };

473 /**
474  * Loops through a passed elements array and calculates a force to apply
475  * to avoid all elements.
476  *
477  * @param {array} elements An array of Flora elements.
478  * @returns {Object} A force to apply.
479  */
480 Agent.prototype.separate = function(elements) {
481 
482   'use strict';
483 
484   var i, max, element, diff, d,
485   sum = new exports.Vector(),
486   count = 0, steer;
487 
488   for (i = 0, max = elements.length; i < max; i += 1) {
489     element = elements[i];
490     if (this.className === element.className && this.id !== element.id) {
491 
492       d = this.location.distance(element.location);
493 
494       if ((d > 0) && (d < this.desiredSeparation)) {
495         diff = exports.Vector.VectorSub(this.location, element.location);
496         diff.normalize();
497         diff.div(d);
498         sum.add(diff);
499         count += 1;
500       }
501     }
502   }
503   if (count > 0) {
504     sum.div(count);
505     sum.normalize();
506     sum.mult(this.maxSpeed);
507     steer = exports.Vector.VectorSub(sum, this.velocity);
508     steer.limit(this.maxSteeringForce);
509     return steer;
510   }
511   return new exports.Vector();
512 };

514 /**
515  * Loops through a passed elements array and calculates a force to apply
516  * to align with all elements.
517  *
518  * @param {array} elements An array of Flora elements.
519  * @returns {Object} A force to apply.
520  */
521 Agent.prototype.align = function(elements) {
522 
523   'use strict';
524 
525   var i, max, element, diff, d,
526     sum = new exports.Vector(),
527     neighbordist = this.width * 2,
528     count = 0, steer;
529 
530   for (i = 0, max = elements.length; i < max; i += 1) {
531     element = elements[i];
532     d = this.location.distance(element.location);
533 
534     if ((d > 0) && (d < neighbordist)) {
535       if (this.className === element.className && this.id !== element.id) {
536         sum.add(element.velocity);
537         count += 1;
538       }
539     }
540   }
541 
542   if (count > 0) {
543     sum.div(count);
544     sum.normalize();
545     sum.mult(this.maxSpeed);
546     steer = exports.Vector.VectorSub(sum, this.velocity);
547     steer.limit(this.maxSteeringForce);
548     return steer;
549   }
550   return new exports.Vector();
551 };

593 /**
594  * Calculates a force to apply to flee a target. The force is the inverse
595  * of the object's maximum speed.
596  *
597  * @param {Object} target The object to flee from.
598  * @returns {Object} A force to apply.
599  */
600 Agent.prototype.flee = function(target) {
601 
602   'use strict';
603 
604   var desiredVelocity = exports.Vector.VectorSub(target.location, this.location); // find vector pointing at target
605 
606   desiredVelocity.normalize(); // reduce to 1
607   desiredVelocity.mult(-this.maxSpeed); // multiply by maxSpeed in opposite direction
608 
609   return desiredVelocity;
610 };

612 /**
613  * Calculates a force to apply to simulate drag on an object.
614  *
615  * @param {Object} target The object that is applying the drag force.
616  * @returns {Object} A force to apply.
617  */
618 Agent.prototype.drag = function(target) {
619 
620   'use strict';
621 
622   var speed = this.velocity.mag(),
623     dragMagnitude = -1 * target.c * speed * speed, // drag magnitude
624     drag = exports.Utils.clone(this.velocity);
625 
626   drag.normalize(); // drag direction
627   drag.mult(dragMagnitude);
628 
629   return drag;
630 };

632 /**
633  * Calculates a force to apply to simulate attraction on an object.
634  *
635  * @param {Object} attractor The attracting object.
636  * @returns {Object} A force to apply.
637  */
638 Agent.prototype.attract = function(attractor) {
639 
640   'use strict';
641 
642   var force = exports.Vector.VectorSub(attractor.location, this.location),
643     distance, strength;
644 
645   distance = force.mag();
646   distance = exports.Utils.constrain(distance, this.width * this.height/8, attractor.width * attractor.height); // min = scale/8 (totally arbitrary); max = scale; the size of the attractor
647   force.normalize();
648   strength = (attractor.G * attractor.mass * this.mass) / (distance * distance);
649   force.mult(strength);
650 
651   return force;
652 };

661 
662   'use strict';

675 /**
676  * Determines if this object is outside the world bounds.
677  *
678  * @param {Object} world The world object.
679  * @returns {boolean} Returns true if the object is outside the world.
680  */
681 Agent.prototype.checkWorldEdges = function(world) {
682 
683   'use strict';
684 
685   var x = this.location.x,
686     y = this.location.y,
687     velocity = this.velocity,
688     desiredVelocity,
689     steer,
690     maxSpeed,
691     check = false,
692     adjusted_loc,
693     diff;
694 
695   // transform origin is at the center of the object
696 
697   if (this.wrapEdges) {
698     if (this.location.x > world.width) {
699       this.location = new exports.Vector(0, this.location.y);
700       diff = new exports.Vector(x - this.location.x, 0); // get the difference bw the initial location and the adjusted location
701       check = true;
702     } else if (this.location.x < 0) {
703       this.location = new exports.Vector(world.width, this.location.y);
704       diff = new exports.Vector(x - this.location.x, 0);
705       check = true;
706     }
707   } else {
708     if (this.avoidEdges) {
709       if (this.location.x < this.avoidEdgesStrength) {
710         maxSpeed = this.maxSpeed;
711       } else if (this.location.x > exports.world.width - this.avoidEdgesStrength) {
712         maxSpeed = -this.maxSpeed;
713       }
714       if (maxSpeed) {
715         desiredVelocity = new exports.Vector(maxSpeed, this.velocity.y),
716         steer = exports.Vector.VectorSub(desiredVelocity, this.velocity);
717         steer.limit(this.maxSteeringForce);
718         this.applyForce(steer);
719       }
720     }
721     if (this.location.x + this.width/2 > world.width) {
722       this.location = new exports.Vector(world.width - this.width/2, this.location.y);
723       diff = new exports.Vector(x - this.location.x, 0); // get the difference bw the initial location and the adjusted location
724       this.velocity.x *= -1 * this.bounciness;
725       check = true;
726      } else if (this.location.x < this.width/2) {
727       this.location = new exports.Vector(this.width/2, this.location.y);
728       diff = new exports.Vector(x - this.location.x, 0);
729       this.velocity.x *= -1 * this.bounciness;
730       check = true;
731     }
732   }
733 
734   ////
735 
736   maxSpeed = null;
737   if (this.wrapEdges) {
738     if (this.location.y > world.height) {
739       this.location = new exports.Vector(this.location.x, 0);
740       diff = new exports.Vector(0, y - this.location.y);
741       check = true;
742     } else if (this.location.y < 0) {
743       this.location = new exports.Vector(this.location.x, world.height);
744       diff = new exports.Vector(0, y - this.location.y);
745       check = true;
746     }
747   } else {
748     if (this.avoidEdges) {
749       if (this.location.y < this.avoidEdgesStrength) {
750         maxSpeed = this.maxSpeed;
751       } else if (this.location.y > exports.world.height - this.avoidEdgesStrength) {
752         maxSpeed = -this.maxSpeed;
753       }
754       if (maxSpeed) {
755         desiredVelocity = new exports.Vector(this.velocity.x, maxSpeed),
756         steer = exports.Vector.VectorSub(desiredVelocity, this.velocity);
757         steer.limit(this.maxSteeringForce);
758         this.applyForce(steer);
759       }
760     }
761     if (this.location.y + this.height/2 > world.height) {
762       this.location = new exports.Vector(this.location.x, world.height - this.height/2);
763       diff = new exports.Vector(0, y - this.location.y);
764       this.velocity.y *= -1 * this.bounciness;
765       check = true;
766       } else if (this.location.y < this.height/2) {
767       this.location = new exports.Vector(this.location.x, this.height/2);
768       diff = new exports.Vector(0, y - this.location.y);
769       this.velocity.y *= -1 * this.bounciness;
770       check = true;
771     }
772   }
773 
774   if (check && this.controlCamera) {
775     world.location.add(diff); // !! do we need this? // add the distance difference to World.location
776   }
777   return check;
778 };
