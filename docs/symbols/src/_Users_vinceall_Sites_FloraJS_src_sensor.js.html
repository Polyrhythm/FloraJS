  2 /**
  3  * Creates a new Sensor object.
  4  *
  5  * @constructor
  6  * @extends Agent
  7  *
  8  * @param {Object} [opt_options] Options.
  9  * @param {string} [opt_options.type = ''] The type of stimulator that can activate this sensor. eg. 'cold', 'heat', 'light', 'oxygen', 'food', 'predator'
 10  * @param {string} [opt_options.behavior = 'LOVE'] The vehicle carrying the sensor will invoke this behavior when the sensor is activated.
 11  * @param {number} [opt_options.sensitivity = 2] The higher the sensitivity, the farther away the sensor will activate when approaching a stimulus.
 12  * @param {number} [opt_options.width = 5] Width.
 13  * @param {number} [opt_options.height = 5] Height.
 14  * @param {number} [opt_options.offsetDistance = 30] The distance from the center of the sensor's parent.
 15  * @param {number} [opt_options.offsetAngle = 0] The angle of rotation around the vehicle carrying the sensor.
 16  * @param {number} [opt_options.opacity = 0.75] Opacity.
 17  * @param {Object} [opt_options.target = null] A stimulator.
 18  * @param {boolean} [opt_options.activated = false] True if sensor is close enough to detect a stimulator.
 19  * @param {boolean} [opt_options.activatedColor = [200, 200, 200]] The color the sensor will display when activated.
 20  */
 21 function Sensor(opt_options) {
 22 
 23   'use strict';
 24 
 25   var options = opt_options || {};
 26 
 27   exports.Agent.call(this, options);
 28 
 29   this.type = options.type || '';
 30   this.behavior = options.behavior || 'LOVE';
 31   this.sensitivity = options.sensitivity === 0 ? 0 : options.sensitivity || 2;
 32   this.width = options.width === 0 ? 0 : options.width || 7;
 33   this.height = options.height === 0 ? 0 : options.height || 7;
 34   this.offsetDistance = options.offsetDistance === 0 ? 0 : options.offsetDistance|| 30;
 35   this.offsetAngle = options.offsetAngle || 0;
 36   this.opacity = options.opacity === 0 ? 0 : options.opacity || 0.75;
 37   this.target = options.target || null;
 38   this.activated = !!options.activated;
 39   this.activatedColor = options.activatedColor || [200, 200, 200];
 40 }
 41 exports.Utils.extend(Sensor, exports.Agent);

 48 /**
 49  * Called every frame, step() updates the instance's properties.
 50  */
 51 Sensor.prototype.step = function() {
 52 
 53   'use strict';
 54 
 55   var check = false, maxSpeed = 10, i, max;
 56 
 57   if (this.type === "heat" && exports.heats.length > 0) {
 58     for (i = 0, max = exports.heats.length; i < max; i += 1) { // heat
 59       if (this.isInside(this, exports.heats[i], this.sensitivity)) {
 60         this.target = exports.heats[i]; // target this stimulator
 61         this.activated = true; // set activation
 62         check = true;
 63 
 64       }
 65     }
 66   } else if (this.type === "cold" && exports.colds.length > 0) {
 67     for (i = 0, max = exports.colds.length; i < max; i += 1) { // cold
 68       if (this.isInside(this, exports.colds[i], this.sensitivity)) {
 69         this.target = exports.colds[i]; // target this stimulator
 70         this.activated = true; // set activation
 71         check = true;
 72       }
 73     }
 74   } else if (this.type === "predator" && exports.predators.length > 0) {
 75     for (i = 0, max = exports.predators.length; i < max; i += 1) { // predator
 76       if (this.isInside(this, exports.predators[i], this.sensitivity)) {
 77         this.target = exports.predators[i]; // target this stimulator
 78         this.activated = true; // set activation
 79         check = true;
 80       }
 81     }
 82   } else if (this.type === "light" && exports.lights.length > 0) {
 83     for (i = 0, max = exports.lights.length; i < max; i += 1) { // light
 84       if (this.isInside(this, exports.lights[i], this.sensitivity)) {
 85         this.target = exports.lights[i]; // target this stimulator
 86         this.activated = true; // set activation
 87         check = true;
 88       }
 89     }
 90   } else if (this.type === "oxygen" && exports.oxygen.length > 0) {
 91     for (i = 0, max = exports.oxygen.length; i < max; i += 1) { // oxygen
 92       if (this.isInside(this, exports.oxygen[i], this.sensitivity)) {
 93         this.target = exports.oxygen[i]; // target this stimulator
 94         this.activated = true; // set activation
 95         check = true;
 96       }
 97     }
 98   } else if (this.type === "food" && exports.food.length > 0) {
 99     for (i = 0, max = exports.food.length; i < max; i += 1) { // food
100       if (this.isInside(this, exports.food[i], this.sensitivity)) {
101         this.target = exports.food[i]; // target this stimulator
102         this.activated = true; // set activation
103         check = true;
104       }
105     }
106   }
107   if (!check) {
108     this.target = null;
109     this.activated = false;
110     this.color = 'transparent';
111   } else {
112     this.color = this.activatedColor;
113   }
114   if (this.afterStep) {
115     this.afterStep.apply(this);
116   }
117 
118 };

120 /**
121  * Returns the force to apply the vehicle when its sensor is activated.
122  *
123  * @param {Object} params A list of properties.
124  * @param {Object} params.agent The vehicle carrying the sensor.
125  */
126 Sensor.prototype.getActivationForce = function(params) {
127 
128   'use strict';
129 
130   var distanceToTarget, m, steer;
131 
132   switch (this.behavior) {
133 
134     /**
135      * Steers toward target
136      */
137     case "AGGRESSIVE":
138       return this.seek(this.target);
139     /**
140      * Steers away from the target
141      */
142     case "COWARD":
143       var forceCoward = this.seek(this.target);
144       return forceCoward.mult(-1);
145     /**
146      * Speeds toward target and keeps moving
147      */
148     case "LIKES":
149       var dvLikes = exports.Vector.VectorSub(this.target.location, this.location);
150       distanceToTarget = dvLikes.mag();
151       dvLikes.normalize();
152 
153       m = distanceToTarget/params.agent.maxSpeed;
154       dvLikes.mult(m);
155 
156       steer = exports.Vector.VectorSub(dvLikes, params.agent.velocity);
157       steer.limit(params.agent.maxSteeringForce * 0.01);
158       return steer;
159     /**
160      * Arrives at target and remains
161      */
162     case "LOVES":
163       var dvLoves = exports.Vector.VectorSub(this.target.location, this.location); // desiredVelocity
164       distanceToTarget = dvLoves.mag();
165       dvLoves.normalize();
166 
167       if (distanceToTarget > this.width) {
168         m = distanceToTarget/params.agent.maxSpeed;
169         dvLoves.mult(m);
170         steer = exports.Vector.VectorSub(dvLoves, params.agent.velocity);
171         steer.limit(params.agent.maxSteeringForce);
172         return steer;
173       }
174       params.agent.velocity = new exports.Vector();
175       params.agent.acceleration = new exports.Vector();
176       return new exports.Vector();
177     /**
178      * Arrives at target but does not stop
179      */
180     case "EXPLORER":
181 
182       var dvExplorer = exports.Vector.VectorSub(this.target.location, this.location);
183       distanceToTarget = dvExplorer.mag();
184       dvExplorer.normalize();
185 
186       m = distanceToTarget/params.agent.maxSpeed;
187       dvExplorer.mult(-m);
188 
189       steer = exports.Vector.VectorSub(dvExplorer, params.agent.velocity);
190       steer.limit(params.agent.maxSteeringForce * 0.01);
191       return steer;
192     /**
193      * Moves in the opposite direction as fast as possible
194      */
195     case "RUN":
196       return this.flee(this.target);
197 
198     case "ACCELERATE":
199       var forceAccel = params.agent.velocity.clone();
200       forceAccel.normalize(); // get direction
201       return forceAccel.mult(params.agent.minSpeed);
202 
203     case "DECELERATE":
204       var forceDecel = params.agent.velocity.clone();
205       forceDecel.normalize(); // get direction
206       return forceDecel.mult(-params.agent.minSpeed);
207 
208     default:
209       return new exports.Vector();
210   }

213 /**
214  * Checks if a sensor can detect a stimulator.
215  *
216  * @param {Object} params The sensor.
217  * @param {Object} container The stimulator.
218  * @param {number} sensitivity The sensor's sensitivity.
219  */
220 Sensor.prototype.isInside = function(item, container, sensitivity) {
221 
222   'use strict';
223 
224   if (item.location.x + item.width/2 > container.location.x - container.width/2 - (sensitivity * container.width) &&
225     item.location.x - item.width/2 < container.location.x + container.width/2 + (sensitivity * container.width) &&
226     item.location.y + item.height/2 > container.location.y - container.height/2 - (sensitivity * container.height) &&
227     item.location.y - item.height/2 < container.location.y + container.height/2 + (sensitivity * container.height)) {
228     return true;
229   }
230   return false;
231 };
