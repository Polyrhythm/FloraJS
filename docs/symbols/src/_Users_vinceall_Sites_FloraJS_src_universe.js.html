  2 /**
  3  * Creates a new Universe.
  4  *
  5  * @constructor
  6  * @param {Object} [opt_options] Options.
  7  * @param {boolean} [opt_options.isPlaying = true] Set to false to suspend the render loop.
  8  * @param {boolean} [opt_options.zSorted = false] Set to true to sort all elements by their zIndex before rendering.
  9  * @param {boolean} [opt_options.showStats = false] Set to true to render stats on startup.
 10  * @param {boolean} [opt_options.isDeviceMotion = false] Set to true add the devicemotion event listener.
 11  *    Typically use with accelerometer equipped devices.
 12  */
 13 function Universe(opt_options) {
 14 
 15   'use strict';
 16 
 17   var i, max, records, me = this,
 18       options = opt_options || {};
 19 
 20   this.isPlaying = options.isPlaying || true;
 21   this.zSorted = !!options.zSorted;
 22   this.showStats = !!options.showStats;
 23   this.isDeviceMotion = !!options.isDeviceMotion;
 24 
 25   /**
 26    * Holds a list of references to worlds
 27    * in the universe.
 28    * @private
 29    */
 30   this._records = [];
 31 
 32   /**
 33    * Holds a reference to the stats display.
 34    * @private
 35    */
 36   this._statsDisplay = null;
 37 
 38   // Events
 39 
 40   // save the current and last mouse position
 41   exports.Utils.addEvent(document.body, 'mousemove', function(e) {
 42     exports.mouse.locLast = exports.mouse.loc.clone();
 43     if (e.pageX && e.pageY) {
 44       exports.mouse.loc = new exports.Vector(e.pageX, e.pageY);
 45     } else if (e.clientX && e.clientY) {
 46       exports.mouse.loc = new exports.Vector(e.clientX, e.clientY);
 47     }
 48   });
 49 
 50   // key input
 51   exports.Utils.addEvent(document, 'keyup', function(e) {
 52     if (e.keyCode === exports.config.keyMap.pause) { // pause
 53       me.pauseSystem();
 54     } else if (e.keyCode === exports.config.keyMap.reset) { // reset system
 55       me.resetSystem();
 56     } else if (e.keyCode === exports.config.keyMap.stats) { // stats
 57       me.toggleStats();
 58     }
 59   });
 60 
 61   // touch input
 62   exports.Utils.addEvent(document, 'touchstart', function(e) {
 63 
 64     var allTouches = e.touches;
 65 
 66     if (allTouches.length === 2) { // stats
 67       me.toggleStats();
 68     } else if (allTouches.length === 3) { // pause
 69       me.pauseSystem();
 70     } else if (allTouches.length === 4) { // reset
 71       me.resetSystem();
 72     }
 73   });
 74 
 75   // key control
 76   exports.Utils.addEvent(document.body, 'keydown', function(e) {
 77     var i, max, elements = exports.elementList.all(),
 78         obj, desired, steer, target,
 79         r, theta, x, y;
 80 
 81     switch(e.keyCode) {
 82       case exports.config.keyMap.thrustLeft:
 83         for(i = 0, max = elements.length; i < max; i++) {
 84           if (elements[i].keyControl) {
 85 
 86             obj = elements[i];
 87 
 88             r = obj.world.width/2; // use angle to calculate x, y
 89             theta = exports.Utils.degreesToRadians(obj.angle - obj.turningRadius);
 90             x = r * Math.cos(theta);
 91             y = r * Math.sin(theta);
 92 
 93             target = exports.Vector.VectorAdd(new exports.Vector(x, y), obj.location);
 94 
 95             desired = exports.Vector.VectorSub(target, obj.location);
 96             desired.normalize();
 97             desired.mult(obj.velocity.mag() * 2);
 98 
 99             steer = exports.Vector.VectorSub(desired, obj.velocity);
100 
101             obj.applyForce(steer);
102           }
103         }
104       break;
105       case exports.config.keyMap.thrustUp:
106         for(i = 0, max = elements.length; i < max; i++) {
107           if (elements[i].keyControl) {
108 
109             obj = elements[i];
110 
111             r = obj.world.width/2;
112             theta = exports.Utils.degreesToRadians(obj.angle);
113             x = r * Math.cos(theta);
114             y = r * Math.sin(theta);
115             desired = new exports.Vector(x, y);
116             desired.normalize();
117             desired.mult(obj.thrust);
118             desired.limit(obj.maxSpeed);
119             elements[i].applyForce(desired);
120           }
121         }
122       break;
123       case exports.config.keyMap.thrustRight:
124         for(i = 0, max = elements.length; i < max; i++) {
125           if (elements[i].keyControl) {
126 
127             obj = elements[i];
128 
129             r = obj.world.width/2; // use angle to calculate x, y
130             theta = exports.Utils.degreesToRadians(obj.angle + obj.turningRadius);
131             x = r * Math.cos(theta);
132             y = r * Math.sin(theta);
133 
134             target = exports.Vector.VectorAdd(new exports.Vector(x, y), obj.location);
135 
136             desired = exports.Vector.VectorSub(target, obj.location);
137             desired.normalize();
138             desired.mult(obj.velocity.mag() * 2);
139 
140             steer = exports.Vector.VectorSub(desired, obj.velocity);
141 
142             elements[i].applyForce(steer);
143           }
144         }
145       break;
146       case exports.config.keyMap.thrustDown:
147         for(i = 0, max = elements.length; i < max; i++) {
148           if (elements[i].keyControl) {
149             elements[i].velocity.mult(0.5);
150           }
151         }
152       break;
153     }
154   });
155 
156   // device motion
157   if (Modernizr.touch && this.isDeviceMotion) {
158     this.addDeviceMotionEventListener();
159   }
160 
161   // stats
162   if (this.showStats) {
163     this.createStats();
164   }
165 }

172 /**
173  * Adds a new World to the 'records' array.
174  *
175  * @returns {Array} An array of elements.
176  */
177 Universe.prototype.addWorld = function(opt_options) {
178 
179   'use strict';
180 
181   var options = opt_options || {};
182 
183   this._records.push(new exports.World(options));
184 
185   // copy reference to new World in elementList
186   exports.elementList.add(this._records[this._records.length - 1]);
187 
188   return this._records;
189 };

194  * @returns {Object} The first world.

196 Universe.prototype.first = function() {
197 
198   'use strict';
199 
200   if (this._records[0]) {

223 /**
224  * Update the properties of a world.
225  *
226  * @param {Object} opt_props A map of properties to update.
227  * @param {string} opt_worldId The id of the world to update. If no id is passed,
228  *    we update the first world in the universe.
229  * @returns {Object} The last world.
230  */
231 Universe.prototype.update = function(opt_props, opt_worldId) {
232 
233   'use strict';
234 
235   var i, props = exports.Interface.getDataType(opt_props) === 'object' ? opt_props : {},
236       world;
237 
238   if (!opt_worldId) {
239     world = this.first();
240   } else {
241     if (exports.Interface.getDataType(opt_worldId) === 'string') {
242       world = this.getWorldById(opt_worldId);
243     } else if (exports.Interface.getDataType(opt_worldId) === 'object') {
244       world = opt_worldId;
245     } else {
246       exports.Utils.log('Universe: update: world param must be null, a string (an id) or reference to a world.');
247       return;
248     }
249   }
250 
251   for (i in props) {
252     if (props.hasOwnProperty(i)) {
253       world[i] = props[i];
254     }
255   }
256 
257   if (props.el) { // if updating the element; update the width and height
258     world.width = parseInt(world.el.style.width.replace('px', ''), 10);
259     world.height = parseInt(world.el.style.height.replace('px', ''), 10);
260   }
261 
262   if (Modernizr.touch && props.isDeviceMotion) {
263     this.addDeviceMotionEventListener();
264   }
265 };

287 /**
288  * Finds an element by its 'id' and returns it.
289  *
290  * @param {string|number} id The element's id.
291  * @returns {Object} The element.
292  */
293 Universe.prototype.getWorldById = function (id) {
294 
295   'use strict';
296 
297   var i, max, records = this._records;
298 
299   for (i = 0, max = records.length; i < max; i += 1) {
300     if (records[i].id === id) {
301       return records[i];
302     }
303   }
304   return null;
305 };
306 
307 /**
308  * Removes a world and its elements.
309  *
310  * @param {string} id The element's id.
311  */
312 Universe.prototype.destroyWorld = function (id) {
313 
314   'use strict';
315 
316   var i, max, records = this._records;
317 
318   for (i = 0, max = records.length; i < max; i += 1) {
319     if (records[i].id === id) {
320 
321       var parent = records[i].el.parentNode;
322 
323       // is this world the body element?
324       if (records[i].el === document.body) {
325         // remove all elements but not the <body>
326         exports.elementList.destroyAll();
327       } else {
328         // remove ell elements and world
329         exports.elementList.destroyByWorld(records[i]);
330         parent.removeChild(records[i].el);
331       }
332       records.splice(i, 1);
333       break;
334     }
335   }
336 };
337 
338 /**
339  * Removes all worlds and resets the 'records' array.
340  */
341 Universe.prototype.destroyAll = function () {
342 
343   'use strict';
344 
345   exports.elementList.destroyAll();
346 
347   for (var i = this._records.length - 1; i >= 0; i -= 1) {
348     this.destroyWorld(this._records[i].id);
349   }
350 };

352 /**
353  * Increments each world's clock.
354  */
355 Universe.prototype.updateClocks = function () {
356 
357   'use strict';
358 
359   for (var i = 0, max = this._records.length; i < max; i += 1) {
360     this._records[i].clock += 1;
361   }
362 };

364 /**
365  * Toggles pausing the FloraSystem animation loop.
366  *
367  * @returns {boolean} True if the system is playing. False if the
368  *    system is not playing.
369  */
370 Universe.prototype.pauseSystem = function() {
371 
372   'use strict';
373 
374   this.isPlaying = !this.isPlaying;
375   if (this.isPlaying) {
376     window.requestAnimFrame(exports.animLoop);
377   }
378   return this.isPlaying;
379 };
380 
381 /**
382  * Resets the FloraSystem.
383  *
384  * @returns {boolean} True if the system is playing. False if the
385  *    system is not playing.
386  */
387 Universe.prototype.resetSystem = function() {
388 
389   'use strict';
390 
391   var i, max, records;
392 
393   // loop thru each world and destroy all elements
394   records = this.all();
395   for (i = 0, max = records.length; i < max; i += 1) {
396     exports.elementList.destroyByWorld(records[i].id);
397   }
398   // call initial setup
399   exports.System.setup();
400   // if system is paused, restart
401   if (!this.isPlaying) {
402     this.isPlaying = true;
403     window.requestAnimFrame(exports.animLoop);
404   }
405 };

407 /**
408  * Toggles StatsDisplay.
409  *
410  * @returns {Object} An instance of StatsDisplay if display is active.
411  *    Null if display is inactive.
412  */
413 Universe.prototype.toggleStats = function() {
414 
415   'use strict';
416 
417   if (!this._statsDisplay) {
418     this.createStats();
419   } else {
420     this.destroyStats();
421   }
422   return this._statsDisplay;
423 };

462 /**
463  * Called from a window devicemotion event, updates the world's gravity
464  * relative to the accelerometer values.
465  * @param {Object} e Event object.
466  */
467 Universe.prototype.devicemotion = function(e) {
468 
469   'use strict';
470 
471   if (window.orientation === 0) {
472     this.update({
473       gravity: new exports.Vector(e.accelerationIncludingGravity.x,
474         e.accelerationIncludingGravity.y * -1)
475     });
476   } else if (window.orientation === -90) {
477     this.update({
478       gravity: new exports.Vector(e.accelerationIncludingGravity.y,
479         e.accelerationIncludingGravity.x )
480     });
481   } else {
482     this.update({
483       gravity: new exports.Vector(e.accelerationIncludingGravity.y * -1,
484         e.accelerationIncludingGravity.x * -1)
485     });
486   }
487 };
