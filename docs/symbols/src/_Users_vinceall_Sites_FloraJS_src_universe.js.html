  2 /**
  3  * Creates a new Universe.
  4  *
  5  * @constructor
  6  * @param {Object} [opt_options] Options.
  7  * @param {boolean} [opt_options.isPlaying = true] Set to false to suspend the render loop.
  8  * @param {boolean} [opt_options.zSorted = false] Set to true to sort all elements by their zIndex before rendering.
  9  * @param {boolean} [opt_options.showStats = false] Set to true to render mr doob stats on startup.
 10  */
 11 function Universe(opt_options) {
 12 
 13   'use strict';
 14 
 15   var i, max, records, me = this,
 16       options = opt_options || {};
 17 
 18   this.isPlaying = options.isPlaying === false ? false : true;
 19   this.zSorted = !!options.zSorted;
 20   this.showStats = !!options.showStats;
 21 
 22   /**
 23    * Holds a list of references to worlds
 24    * in the universe.
 25    * @private
 26    */
 27   this._records = [];
 28 
 29   /**
 30    * Holds a reference to the stats display.
 31    * @private
 32    */
 33   this._statsDisplay = null;
 34 
 35   // Events
 36 
 37   // save the current and last mouse position
 38   exports.Utils.addEvent(document.body, 'mousemove', function(e) {
 39     exports.mouse.locLast = exports.mouse.loc.clone();
 40     if (e.pageX && e.pageY) {
 41       exports.mouse.loc = new exports.Vector(e.pageX, e.pageY);
 42     } else if (e.clientX && e.clientY) {
 43       exports.mouse.loc = new exports.Vector(e.clientX, e.clientY);
 44     }
 45   });
 46 
 47   // toggle the world playstate
 48   exports.Utils.addEvent(document, 'keyup', function(e) {
 49     if (e.keyCode === exports.config.keyMap.toggleWorldPlaystate) { // pause
 50       me.isPlaying = !me.isPlaying;
 51       if (me.isPlaying) {
 52         window.requestAnimFrame(exports.animLoop);
 53       }
 54     } else if (e.keyCode === exports.config.keyMap.resetSystem) { // reset system
 55         // loop thru each world and destroy all elements
 56         records = me.all();
 57         for (i = 0, max = records.length; i < max; i += 1) {
 58           exports.elementList.destroyByWorld(records[i].id);
 59         }
 60         // call initial setup
 61         exports.System.setup();
 62         // if system is pause, restart
 63         if (!me.isPlaying) {
 64           me.isPlaying = true;
 65           window.requestAnimFrame(exports.animLoop);
 66         }
 67     } else if (e.keyCode === exports.config.keyMap.toggleStatsDisplay) { // stats
 68       if (!me._statsDisplay) {
 69         me.createStats();
 70       } else {
 71         me.destroyStats();
 72       }
 73     }
 74   });
 75 
 76   // key control
 77   exports.Utils.addEvent(document.body, 'keydown', function(e) {
 78     var i, max, elements = exports.elementList.all(),
 79         obj, desired, steer, target,
 80         r, theta, x, y;
 81 
 82     switch(e.keyCode) {
 83       case exports.config.keyMap.thrustLeft:
 84         for(i = 0, max = elements.length; i < max; i++) {
 85           if (elements[i].keyControl) {
 86 
 87             obj = elements[i];
 88 
 89             r = obj.world.width/2; // use angle to calculate x, y
 90             theta = exports.Utils.degreesToRadians(obj.angle - obj.turningRadius);
 91             x = r * Math.cos(theta);
 92             y = r * Math.sin(theta);
 93 
 94             target = exports.Vector.VectorAdd(new exports.Vector(x, y), obj.location);
 95 
 96             desired = exports.Vector.VectorSub(target, obj.location);
 97             desired.normalize();
 98             desired.mult(obj.velocity.mag() * 2);
 99 
100             steer = exports.Vector.VectorSub(desired, obj.velocity);
101 
102             obj.applyForce(steer);
103           }
104         }
105       break;
106       case exports.config.keyMap.thrustUp:
107         for(i = 0, max = elements.length; i < max; i++) {
108           if (elements[i].keyControl) {
109 
110             obj = elements[i];
111 
112             r = obj.world.width/2;
113             theta = exports.Utils.degreesToRadians(obj.angle);
114             x = r * Math.cos(theta);
115             y = r * Math.sin(theta);
116             desired = new exports.Vector(x, y);
117             desired.normalize();
118             desired.mult(obj.thrust);
119             desired.limit(obj.maxSpeed);
120             elements[i].applyForce(desired);
121           }
122         }
123       break;
124       case exports.config.keyMap.thrustRight:
125         for(i = 0, max = elements.length; i < max; i++) {
126           if (elements[i].keyControl) {
127 
128             obj = elements[i];
129 
130             r = obj.world.width/2; // use angle to calculate x, y
131             theta = exports.Utils.degreesToRadians(obj.angle + obj.turningRadius);
132             x = r * Math.cos(theta);
133             y = r * Math.sin(theta);
134 
135             target = exports.Vector.VectorAdd(new exports.Vector(x, y), obj.location);
136 
137             desired = exports.Vector.VectorSub(target, obj.location);
138             desired.normalize();
139             desired.mult(obj.velocity.mag() * 2);
140 
141             steer = exports.Vector.VectorSub(desired, obj.velocity);
142 
143             elements[i].applyForce(steer);
144           }
145         }
146       break;
147       case exports.config.keyMap.thrustDown:
148         for(i = 0, max = elements.length; i < max; i++) {
149           if (elements[i].keyControl) {
150             elements[i].velocity.mult(0.5);
151           }
152         }
153       break;
154     }
155   });
156 
157   if (this.showStats) {
158     this.createStats();
159   }
160 }

167 /**
168  * Adds a new World to the 'records' array.
169  *
170  * @returns {Array} An array of elements.
171  */
172 Universe.prototype.addWorld = function(opt_options) {
173 
174   'use strict';
175 
176   var options = opt_options || {};
177 
178   this._records.push(new exports.World(options));
179 
180   // copy reference to new World in elementList
181   exports.elementList.add(this._records[this._records.length - 1]);
182 
183   return this._records;
184 };

186 /**
187  * Returns the first item in 'records' array.
188  *
189  * @returns {Object} The first world.
190  */
191 Universe.prototype.first = function() {
192 
193   'use strict';
194 
195   if (this._records[0]) {
196     return this._records[0];
197   } else {
198     return null;
199   }

213   } else {

218 /**
219  * Update the properties of a world.
220  *
221  * @returns {Object} The last world.
222  */
223 Universe.prototype.update = function(opt_props, opt_world) {
224 
225   'use strict';
226 
227   var i, props = exports.Interface.getDataType(opt_props) === 'object' ? opt_props : {},
228       world;
229 
230   if (!opt_world) {
231     world = this.first();
232   } else {
233     if (exports.Interface.getDataType(opt_world) === 'string') {
234       world = this.getWorldById(opt_world);
235     } else if (exports.Interface.getDataType(opt_world) === 'object') {
236       world = opt_world;
237     } else {
238       exports.Utils.log('Universe: update: world param must be null, a string (an id) or reference to a world.');
239       return;
240     }
241   }
242 
243   for (i in props) {
244     if (props.hasOwnProperty(i)) {
245       world[i] = props[i];
246     }
247   }
248 
249   if (props.el) { // if updating the element; update the width and height
250     world.width = parseInt(world.el.style.width.replace('px', ''), 10);
251     world.height = parseInt(world.el.style.height.replace('px', ''), 10);
252   }
253 };

275 /**
276  * Finds an element by its 'id' and returns it.
277  *
278  * @param {string|number} id The element's id.
279  * @returns {Object} The element.
280  */
281 Universe.prototype.getWorldById = function (id) {
282 
283   'use strict';
284 
285   var i, max, records = this._records;
286 
287   for (i = 0, max = records.length; i < max; i += 1) {
288     if (records[i].id === id) {
289       return records[i];
290     }
291   }
292   return null;
293 };

295 /**
296  * Removes a world and its elements.
297  *
298  * @param {string} id The element's id.
299  */
300 Universe.prototype.destroyWorld = function (id) {
301 
302   'use strict';
303 
304   var i, max, records = this._records;
305 
306   for (i = 0, max = records.length; i < max; i += 1) {
307     if (records[i].id === id) {
308 
309       var parent = records[i].el.parentNode;
310 
311       // is this world the body element?
312       if (records[i].el === document.body) {
313         // remove all elements but not the <body>
314         exports.elementList.destroyAll();
315       } else {
316         // remove ell elements and world
317         exports.elementList.destroyByWorld(records[i]);
318         parent.removeChild(records[i].el);
319       }
320       records.splice(i, 1);
321       break;
322     }
323   }
324 };
325 
326 /**
327  * Removes all worlds and resets the 'records' array.
328  */
329 Universe.prototype.destroyAll = function () {
330 
331   'use strict';
332 
333   exports.elementList.destroyAll();
334 
335   for (var i = this._records.length - 1; i >= 0; i -= 1) {
336     this.destroyWorld(this._records[i].id);
337   }
338 };

340 /**
341  * Increments each world's clock.
342  */
343 Universe.prototype.updateClocks = function () {
344 
345   'use strict';
346 
347   for (var i = 0, max = this._records.length; i < max; i += 1) {
348     this._records[i].clock += 1;
349   }
350 };
