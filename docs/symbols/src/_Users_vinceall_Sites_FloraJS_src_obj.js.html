  2 /**
  3  * Creates a new Obj. All Flora elements extend Obj.
  4  * @constructor
  5  */
  6 function Obj() {
  7 
  8   'use strict';
  9 
 10   var i, max, key, prop;
 11 
 12   for (i = 0, max = arguments.length; i < max; i += 1) {
 13     prop = arguments[i];
 14     for (key in prop) {
 15       if (prop.hasOwnProperty(key)) {
 16         this[key] = prop[key];
 17       }
 18     }
 19   }
 20 }
 21 
 22 Obj.events =[
 23   "mouseenter",
 24   "mousedown",
 25   "mousemove",
 26   "mouseup",
 27   "mouseleave"
 28 ];
 29 
 30 /**
 31  * Define a name property.
 32  */
 33 Obj.prototype.name = 'obj';

 35 /**
 36  * Called by a mouseover event listener.
 37  *
 38  * @param {Object} e The event object passed by the listener.
 39  */
 40 Obj.mouseover = function(e) {
 41 
 42   'use strict';
 43 
 44   this.isMouseOut = false;
 45   clearInterval(this.mouseOutInterval);
 46 };
 47 
 48 /**
 49  * Called by a mousedown event listener.
 50  *
 51  * @param {Object} e The event object passed by the listener.
 52  */
 53 Obj.mousedown = function(e) {
 54 
 55   'use strict';
 56 
 57   var target = e.target;
 58 
 59   this.isPressed = true;
 60   this.isMouseOut = false;
 61   this.offsetX = e.pageX - target.offsetLeft;
 62   this.offsetY = e.pageY - target.offsetTop;
 63 };
 64 
 65 /**
 66  * Called by a mousemove event listener.
 67  *
 68  * @param {Object} e The event object passed by the listener.
 69  */
 70 Obj.mousemove = function(e) {
 71 
 72   'use strict';
 73 
 74   var x, y;
 75 
 76   if (this.isPressed) {
 77 
 78     this.isMouseOut = false;
 79 
 80     x = e.pageX - this.world.el.offsetLeft;
 81     y = e.pageY - this.world.el.offsetTop;
 82 
 83     if (Obj.mouseIsInsideWorld(this.world)) {
 84       this.location = new exports.Vector(x, y);
 85     } else {
 86       this.isPressed = false;
 87     }
 88   }
 89 };
 90 
 91 /**
 92  * Called by a mouseup event listener.
 93  *
 94  * @param {Object} e The event object passed by the listener.
 95  */
 96 Obj.mouseup = function(e) {
 97 
 98   'use strict';
 99 
100   this.isPressed = false;
101 };
102 
103 /**
104  * Called by a mouseout event listener.
105  *
106  * @param {Object} e The event object passed by the listener.
107  */
108 Obj.mouseout = function(e) {
109 
110   'use strict';
111 
112   var me = this,
113     x, y;
114 
115   if (this.isPressed) {
116 
117     this.isMouseOut = true;
118 
119     this.mouseOutInterval = setInterval(function () { // if mouse is too fast for block update, update via an interval until it catches up
120 
121       if (me.isPressed && me.isMouseOut) {
122 
123         x = exports.mouse.loc.x - me.world.el.offsetLeft;
124         y = exports.mouse.loc.y - me.world.el.offsetTop;
125 
126         me.location = new exports.Vector(x, y);
127       }
128     }, 16);
129   }
130 };
131 
132 /**
133  * Checks if mouse location is inside the world.
134  *
135  * @param {Object} world A Flora world.
136  * @returns {boolean} True if mouse is inside world; False if
137  *    mouse is outside world.
138  */
139 Obj.mouseIsInsideWorld = function(world) {
140 
141   'use strict';
142 
143   var x = exports.mouse.loc.x,
144       y = exports.mouse.loc.y;
145 
146   if (world) {
147     if (x > world.location.x &&
148       x < world.location.x + world.width &&
149       y > world.location.y &&
150       y < world.location.y + world.height) {
151       return true;
152     }
153   }
154   return false;
155 };

157 /**
158  * Renders the element to the DOM. Called every frame.
159  */
160 Obj.prototype.draw = function() {
161 
162   'use strict';
163 
164   this.el.style.cssText = exports.Utils.getCSSText({
165     x: this.location.x - this.width/2,
166     y: this.location.y - this.height/2,
167     s: this.scale,
168     a: this.angle,
169     o: this.opacity,
170     w: this.width,
171     h: this.height,
172     cm: this.colorMode,
173     color: this.color,
174     z: this.zIndex,
175     borderWidth: this.borderWidth,
176     borderStyle: this.borderStyle,
177     borderColor: this.borderColor,
178     borderRadius: this.borderRadius,
179     boxShadow: this.boxShadow
180   });
181 };
