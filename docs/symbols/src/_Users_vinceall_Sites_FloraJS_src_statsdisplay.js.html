1 /*global exports, document, window */
  2 /**
  3  * Creates a new StatsDisplay object.
  4  *
  5  * Use this class to create a field in the
  6  * top-left corner that displays the current
  7  * frames per second and total number of elements
  8  * processed in the System.animLoop.
  9  *
 10  * Note: StatsDisplay will not function in browsers
 11  * whose Date object does not support Date.now().
 12  * These include IE6, IE7, and IE8.
 13  *
 14  * @constructor
 15  */
 16 function StatsDisplay() {
 17 
 18   'use strict';
 19 
 20   var labelContainer, label;
 21 
 22   /**
 23    * Frames per second.
 24    * @private
 25    */
 26   this._fps = 0;
 27 
 28   /**
 29    * The current time.
 30    * @private
 31    */
 32   if (Date.now) {
 33     this._time = Date.now();
 34   } else {
 35     this._time = 0;
 36   }
 37 
 38   /**
 39    * The time at the last frame.
 40    * @private
 41    */
 42   this._timeLastFrame = this._time;
 43 
 44   /**
 45    * The time the last second was sampled.
 46    * @private
 47    */
 48   this._timeLastSecond = this._time;
 49 
 50   /**
 51    * Holds the total number of frames
 52    * between seconds.
 53    * @private
 54    */
 55   this._frameCount = 0;
 56 
 57   /**
 58    * A reference to the DOM element containing the display.
 59    * @private
 60    */
 61   this._el = document.createElement('div');
 62   this._el.id = 'statsDisplay';
 63   this._el.className = 'statsDisplay';
 64   this._el.style.color = 'white';
 65 
 66   /**
 67    * A reference to the textNode displaying the total number of elements.
 68    * @private
 69    */
 70   this._totalElementsValue = null;
 71 
 72   /**
 73    * A reference to the textNode displaying the frame per second.
 74    * @private
 75    */
 76   this._fpsValue = null;
 77 
 78   // create 3dTransforms label
 79   labelContainer = document.createElement('span');
 80   labelContainer.className = 'statsDisplayLabel';
 81   label = document.createTextNode('trans3d: ');
 82   labelContainer.appendChild(label);
 83   this._el.appendChild(labelContainer);
 84 
 85   // create textNode for totalElements
 86   this._3dTransformsValue = document.createTextNode(exports.System.supportedFeatures.csstransforms3d);
 87   this._el.appendChild(this._3dTransformsValue);
 88 
 89   // create totol elements label
 90   labelContainer = document.createElement('span');
 91   labelContainer.className = 'statsDisplayLabel';
 92   label = document.createTextNode('total elements: ');
 93   labelContainer.appendChild(label);
 94   this._el.appendChild(labelContainer);
 95 
 96   // create textNode for totalElements
 97   this._totalElementsValue = document.createTextNode('0');
 98   this._el.appendChild(this._totalElementsValue);
 99 
100   // create fps label
101   labelContainer = document.createElement('span');
102   labelContainer.className = 'statsDisplayLabel';
103   label = document.createTextNode('fps: ');
104   labelContainer.appendChild(label);
105   this._el.appendChild(labelContainer);
106 
107   // create textNode for fps
108   this._fpsValue = document.createTextNode('0');
109   this._el.appendChild(this._fpsValue);
110 
111   document.body.appendChild(this._el);
112 
113   /**
114    * Initiates the requestAnimFrame() loop.
115    */
116   this._update(this);
117 }
118 
119 /**
120  * Returns the current frames per second value.
121  * @returns {number} Frame per second.
122  */
123 StatsDisplay.prototype.getFPS = function() {
124 
125   'use strict';
126 
127   return this._fps;
128 };
129 
130 /**
131  * If 1000ms have elapsed since the last evaluated second,
132  * _fps is assigned the total number of frames rendered and
133  * its corresponding textNode is updated. The total number of
134  * elements is also updated.
135  *
136  * This function is called again via requestAnimFrame().
137  *
138  * @private
139  */
140 StatsDisplay.prototype._update = function(me) {
141 
142   'use strict';
143 
144   var elementCount = exports.elementList.count();
145 
146   if (Date.now) {
147     me._time = Date.now();
148   } else {
149     me._time = 0;
150   }
151   me._frameCount++;
152 
153   // at least a second has passed
154   if (me._time > me._timeLastSecond + 1000) {
155 
156     me._fps = me._frameCount;
157     me._timeLastSecond = me._time;
158     me._frameCount = 0;
159 
160     me._fpsValue.nodeValue = me._fps;
161     me._totalElementsValue.nodeValue = elementCount;
162     me._3dTransformsValue.nodeValue = exports.System.supportedFeatures.csstransforms3d;
163   }
164 
165   var reqAnimFrame = (function (me) {
166     return (function() {
167       me._update(me);
168     });
169   })(this);
170 
171   window.requestAnimFrame(reqAnimFrame);
172 };
173 
174 StatsDisplay.prototype.name = 'StatsDisplay';
175 
176 exports.StatsDisplay = StatsDisplay;
