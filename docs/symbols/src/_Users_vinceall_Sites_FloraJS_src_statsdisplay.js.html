  2 /**
  3  * Creates a new StatsDisplay object.
  4  *
  5  * Use this class to create a field in the
  6  * top-left corner that displays the current
  7  * frames per second and total number of
  8  * processed in the System.animLoop.
  9  *
 10  * @constructor
 11  */
 12 function StatsDisplay() {
 13 
 14   'use strict';
 15 
 16   var labelContainer, label;
 17 
 18   if (!Date.now) {
 19     return;
 20   }
 21 
 22   /**
 23    * Frames per second.
 24    * @private
 25    */
 26   this._fps = 0;
 27 
 28   /**
 29    * The current time.
 30    * @private
 31    */
 32   this._time = Date.now();
 33 
 34   /**
 35    * The time at the last frame.
 36    * @private
 37    */
 38   this._timeLastFrame = this._time;
 39 
 40   /**
 41    * The time the last second was sampled.
 42    * @private
 43    */
 44   this._timeLastSecond = this._time;
 45 
 46   /**
 47    * Holds the total number of frames
 48    * between seconds.
 49    * @private
 50    */
 51   this._frameCount = 0;
 52 
 53   /**
 54    * A reference to the DOM element containing the display.
 55    * @private
 56    */
 57   this._el = document.createElement('div');
 58   this._el.id = 'statsDisplay';
 59   this._el.className = 'statsDisplay';
 60   this._el.style.color = 'white';
 61 
 62   /**
 63    * A reference to the textNode displaying the total number of elements.
 64    * @private
 65    */
 66   this._totalElementsValue = null;
 67 
 68   /**
 69    * A reference to the textNode displaying the frame per second.
 70    * @private
 71    */
 72   this._fpsValue = null;
 73 
 74   // create totol elements label
 75   labelContainer = document.createElement('span');
 76   labelContainer.className = 'statsDisplayLabel';
 77   label = document.createTextNode('total elements: ');
 78   labelContainer.appendChild(label);
 79   this._el.appendChild(labelContainer);
 80 
 81   // create textNode for totalElements
 82   this._totalElementsValue = document.createTextNode('0');
 83   this._el.appendChild(this._totalElementsValue);
 84 
 85   // create fps label
 86   labelContainer = document.createElement('span');
 87   labelContainer.className = 'statsDisplayLabel';
 88   label = document.createTextNode('fps: ');
 89   labelContainer.appendChild(label);
 90   this._el.appendChild(labelContainer);
 91 
 92   // create textNode for fps
 93   this._fpsValue = document.createTextNode('0');
 94   this._el.appendChild(this._fpsValue);
 95 
 96   document.body.appendChild(this._el);
 97 
 98   /**
 99    * Initiates the requestAnimFrame() loop.
100    */
101   this._update();
102 }
103 
104 /**
105  * Returns the current frames per second value.
106  * @returns {number} Frame per second.
107  */
108 StatsDisplay.prototype.getFPS = function() {
109 
110   'use strict';
111 
112   return this._fps;
113 };
114 
115 /**
116  * If 1000ms have elapsed since the last evaluated second,
117  * _fps is assigned the total number of frames rendered and
118  * its corresponding textNode is updated. The total number of
119  * elements is also updated.
120  *
121  * This function is called again via requestAnimFrame().
122  *
123  * @private
124  */
125 StatsDisplay.prototype._update = function() {
126 
127   'use strict';
128 
129   var elementCount = exports.elementList.count();
130 
131   this._time = Date.now();
132   this._frameCount++;
133 
134   // at least a second has passed
135   if (this._time > this._timeLastSecond + 1000) {
136 
137     this._fps = this._frameCount;
138     this._timeLastSecond = this._time;
139     this._frameCount = 0;
140 
141     this._fpsValue.nodeValue = this._fps;
142     this._totalElementsValue.nodeValue = elementCount;
143   }
144   window.requestAnimFrame(this._update.bind(this));
145 };
146 
147 /**
148  * Define a name property.
149  */
150 StatsDisplay.prototype.name = 'statsDisplay';
151 
152 exports.StatsDisplay = StatsDisplay;
