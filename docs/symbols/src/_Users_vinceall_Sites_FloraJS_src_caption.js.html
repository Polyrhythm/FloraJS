  2 /**
  3  * Creates a new Caption object.
  4  * Use captions to communicate short messages to users like a title
  5  * or simple instructions like 'click for more particles'.
  6  *
  7  * @constructor
  8  *
  9  * @param {Object} [opt_options] Options.
 10  * @param {string} [opt_options.position = 'top left'] A text representation
 11  *    of the caption's location. Possible values are 'top left', 'top center', 'top right',
 12  *    'bottom left', 'bottom center', 'bottom right', 'center'.
 13  * @param {string} [opt_options.text = ''] The caption's text.
 14  * @param {number} [opt_options.opacity = 0.75] The caption's opacity.
 15  * @param {number} [opt_options.color = [255, 255, 255]] The caption's color.
 16  * @param {string} [opt_options.borderWidth = '1px'] The caption's border width.
 17  * @param {string} [opt_options.borderStyle = 'solid'] The caption's border style.
 18  * @param {Array|string} [opt_options.borderColor = 0.75] The caption's border color.
 19  */
 20 function Caption(opt_options) {
 21 
 22   'use strict';
 23 
 24   var options = opt_options || {};
 25 
 26   // if a world is not passed, use the first world in the universe
 27   this.world = options.world || exports.universe.first();
 28   this.position = options.position || 'top left';
 29   this.text = options.text || '';
 30   this.opacity = options.opacity === 0 ? 0 : options.opacity || 0.75;
 31   this.color = options.color || [255, 255, 255];
 32   this.borderWidth = options.borderWidth || '1px';
 33   this.borderStyle = options.borderStyle || 'solid';
 34   this.borderColor = options.borderColor || [204, 204, 204];
 35   this.colorMode = options.colorMode || 'rgb';
 36 
 37   /**
 38    * Holds a reference to the caption's DOM elements.
 39    * @private
 40    */
 41   this._el = document.createElement('div');
 42   this._el.id = 'caption';
 43   this._el.className = 'caption ' + this.position;
 44   this._el.style.opacity = this.opacity;
 45   this._el.style.color = this.colorMode + '(' + this.color[0] + ', ' + this.color[1] +
 46         ', ' + this.color[2] + ')';
 47   this._el.style.borderWidth = this.borderWidth;
 48   this._el.style.borderStyle = this.borderStyle;
 49   if (typeof this.borderColor === 'string') {
 50     this._el.style.borderColor = this.borderColor;
 51   } else {
 52     this._el.style.borderColor = this.colorMode + '(' + this.borderColor[0] + ', ' + this.borderColor[1] +
 53         ', ' + this.borderColor[2] + ')';
 54   }
 55   this._el.appendChild(document.createTextNode(this.text));
 56   if (document.getElementById('caption')) {
 57     document.getElementById('caption').parentNode.removeChild(document.getElementById('caption'));
 58   }
 59   this.world.el.appendChild(this._el);
 60 }
 61 
 62 Caption.prototype.name = 'Caption';
 63 
 64 /**
 65  * Removes the caption's DOM element.
 66  */
 67 Caption.prototype.destroy = function() {
 68 
 69   'use strict';
 70 
 71   this._el.parentNode.removeChild(this._el);
 72 };
 73 
 74 exports.Caption = Caption;
