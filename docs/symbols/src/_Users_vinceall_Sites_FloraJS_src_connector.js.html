  2 /**
  3  * Creates a new Connector.
  4  *
  5  * @constructor
  6  * @extends Agent
  7  * @param {Object} parentA The object that starts the connection.
  8  * @param {Object} parentB The object that ends the connection.
  9  * @param {Object} [opt_options] Options.
 10  * @param {number} [opt_options.opacity = 0.25] Opacity.
 11  * @param {number} [opt_options.zIndex = 0] zIndex.
 12  */
 13 function Connector(parentA, parentB, opt_options) {
 14 
 15   'use strict';
 16 
 17   var options = opt_options || {};
 18 
 19   exports.Agent.call(this, options);
 20 
 21   if (!parentA || !parentB) {
 22     throw new Error('Connector: both parentA and parentB are required.');
 23   }
 24   this.parentA = parentA;
 25   this.parentB = parentB;
 26   this.width = 0;
 27   this.height = 0;
 28   this.color = 'transparent';
 29 
 30   this.opacity = options.opacity === 0 ? 0 : options.opacity || 1;
 31   this.zIndex = options.zIndex || 0;
 32 
 33 }
 34 exports.Utils.extend(Connector, exports.Agent);
 35 
 36 /**
 37  * Define a name property. Used to assign a class name and prefix an id.
 38  */
 39 Connector.prototype.name = 'connector';
 40 
 41 /**
 42  * Called every frame, step() updates the instance's properties.
 43  */
 44 Connector.prototype.step = function() {
 45 
 46   'use strict';
 47 
 48   var a = this.parentA.location,
 49       b = this.parentB.location;
 50 
 51   this.width = Math.floor(exports.Vector.VectorSub(this.parentA.location, this.parentB.location).mag());
 52   this.location = exports.Vector.VectorAdd(this.parentA.location, this.parentB.location).div(2); // midpoint = (v1 + v2)/2
 53   this.angle = exports.Utils.radiansToDegrees(Math.atan2(b.y - a.y, b.x - a.x) );
 54 };
 55 
 56 exports.Connector = Connector;
