  2 /**
  3  * Creates a new Walker.
  4  *
  5  * @constructor
  6  * @extends Agent
  7  *
  8  * @param {Object} [opt_options] Walker options.
  9  * @param {string} [opt_options.className = 'walker'] The corresponding DOM element's class name.
 10  * @param {boolean} [opt_options.isPerlin = true] If set to true, object will use Perlin Noise to calculate its location.
 11  * @param {boolean} [opt_options.remainsOnScreen = false] If set to true and isPerlin = true, object will avoid world edges.
 12  * @param {number} [opt_options.perlinSpeed = 0.005] If isPerlin = true, perlinSpeed determines how fast the object location moves through the noise space.
 13  * @param {number} [opt_options.perlinTime = 0] Sets the Perlin Noise time.
 14  * @param {number} [opt_options.perlinAccelLow = -0.075] The lower bound of acceleration when isPerlin = true.
 15  * @param {number} [opt_options.perlinAccelHigh = 0.075] The upper bound of acceleration when isPerlin = true.
 16  * @param {number} [opt_options.offsetX = Math.random() * 10000] The x offset in the Perlin Noise space.
 17  * @param {number} [opt_options.offsetY = Math.random() * 10000] The y offset in the Perlin Noise space.
 18  * @param {boolean} [opt_options.isRandom = false] Set to true for walker to move in a random direction.
 19  * @param {number} [opt_options.randomRadius = 100] If isRandom = true, walker will look for a new location each frame based on this radius.
 20  * @param {boolean} [opt_options.isHarmonic = false] If set to true, walker will move using harmonic motion.
 21  * @param {Object} [opt_options.isHarmonic = {x: 6, y: 6}] If isHarmonic = true, sets the motion's amplitude.
 22  * @param {Object} [opt_options.harmonicPeriod = {x: 150, y: 150}] If isHarmonic = true, sets the motion's period.
 23  * @param {number} [opt_options.width = 10] Width
 24  * @param {number} [opt_options.height = 10] Height
 25  * @param {number} [opt_options.maxSpeed = 30] Maximum speed
 26  * @param {boolean} [opt_options.wrapEdges = false] Set to true to set the object's location to the opposite side of the world if the object moves outside the world's bounds.
 27  * @param {boolean} [opt_options.isStatic = false] If true, object will not move.
 28  */
 29 function Walker(opt_options) {
 30 
 31   'use strict';
 32 
 33   var options = opt_options || {};
 34 
 35   exports.Agent.call(this, options);
 36 
 37   this.isPerlin = options.isPerlin === false ? false : options.isPerlin || true;
 38   this.remainsOnScreen = !!options.remainsOnScreen;
 39   this.perlinSpeed = options.perlinSpeed || 0.005;
 40   this.perlinTime = options.perlinTime || 0;
 41   this.perlinAccelLow = options.perlinAccelLow || -0.075;
 42   this.perlinAccelHigh = options.perlinAccelHigh || 0.075;
 43   this.offsetX = options.offsetX || Math.random() * 10000;
 44   this.offsetY = options.offsetY || Math.random() * 10000;
 45   this.isRandom = !!options.isRandom;
 46   this.randomRadius = options.randomRadius || 100;
 47   this.isHarmonic = !!options.isHarmonic;
 48   this.harmonicAmplitude = options.harmonicAmplitude || new exports.Vector(4, 0);
 49   this.harmonicPeriod = options.harmonicPeriod || new exports.Vector(300, 1);
 50   this.width = options.width === 0 ? 0 : options.width || 10;
 51   this.height = options.height === 0 ? 0 : options.height || 10;
 52   this.maxSpeed = options.maxSpeed === 0 ? 0 : options.maxSpeed || 30;
 53   this.wrapEdges = !!options.wrapEdges;
 54   this.isStatic = !!options.isStatic;
 55 }
 56 exports.Utils.extend(Walker, exports.Agent);
 57 
 58 Walker.prototype.name = 'Walker';
 59 
 60 /**
 61  * Called every frame, step() updates the instance's properties.
 62  */
 63 Walker.prototype.step = function () {
 64 
 65   'use strict';
 66 
 67   var world = this.world,
 68       friction;
 69 
 70   if (this.beforeStep) {
 71     this.beforeStep.apply(this);
 72   }
 73 
 74   if (!this.isStatic && !this.isPressed) {
 75 
 76     if (this.isPerlin) {
 77 
 78       this.perlinTime += this.perlinSpeed;
 79 
 80       if (this.remainsOnScreen) {
 81         this.acceleration = new exports.Vector();
 82         this.velocity = new exports.Vector.create();
 83         this.location.x =  exports.Utils.map(exports.SimplexNoise.noise(this.perlinTime + this.offsetX, 0, 0.1), -1, 1, 0, exports.world.width);
 84         this.location.y =  exports.Utils.map(exports.SimplexNoise.noise(0, this.perlinTime + this.offsetY, 0.1), -1, 1, 0, exports.world.height);
 85       } else {
 86         this.acceleration.x =  exports.Utils.map(exports.SimplexNoise.noise(this.perlinTime + this.offsetX, 0, 0.1), -1, 1, this.perlinAccelLow, this.perlinAccelHigh);
 87         this.acceleration.y =  exports.Utils.map(exports.SimplexNoise.noise(0, this.perlinTime + this.offsetY, 0.1), -1, 1, this.perlinAccelLow, this.perlinAccelHigh);
 88       }
 89 
 90     } else {
 91 
 92       // start -- APPLY FORCES
 93 
 94       if (world.c) { // friction
 95         friction = exports.Utils.clone(this.velocity);
 96         friction.mult(-1);
 97         friction.normalize();
 98         friction.mult(world.c);
 99         this.applyForce(friction);
100       }
101 
102       this.applyForce(world.wind); // wind
103       this.applyForce(world.gravity); // gravity
104     }
105 
106     if (this.isHarmonic) {
107       this.velocity.x = this.harmonicAmplitude.x * Math.cos((Math.PI * 2) * exports.world.clock / this.harmonicPeriod.x);
108       this.velocity.y = this.harmonicAmplitude.y * Math.cos((Math.PI * 2) * exports.world.clock / this.harmonicPeriod.y);
109     }
110 
111     if (this.isRandom) {
112       this.target = { // find a random point and steer toward it
113         location: exports.Vector.VectorAdd(this.location, new exports.Vector(exports.Utils.getRandomNumber(-this.randomRadius, this.randomRadius), exports.Utils.getRandomNumber(-this.randomRadius, this.randomRadius)))
114       };
115     }
116 
117     if (this.seekTarget) { // follow seek target
118       this.applyForce(this.seek(this.seekTarget));
119     }
120 
121     // end -- APPLY FORCES
122 
123     this.velocity.add(this.acceleration); // add acceleration
124 
125     if (this.maxSpeed) {
126       this.velocity.limit(this.maxSpeed); // check if velocity > maxSpeed
127     }
128 
129     this.location.add(this.velocity); // add velocity
130 
131     if (this.pointToDirection) { // object rotates toward direction
132       if (this.velocity.mag() > 0.1) { // rotate toward direction?
133         this.angle = exports.Utils.radiansToDegrees(Math.atan2(this.velocity.y, this.velocity.x));
134       }
135     }
136 
137     if (this.controlCamera) { // check camera after velocity calculation
138       this.checkCameraEdges();
139     }
140 
141     if (this.checkEdges || this.wrapEdges) {
142       this.checkWorldEdges(world);
143     }
144 
145     if (this.lifespan > 0) {
146       this.lifespan -= 1;
147     }
148 
149     if (this.afterStep) {
150       this.afterStep.apply(this);
151     }
152 
153     this.acceleration.mult(0); // reset acceleration
154   }
155 };
156 exports.Walker = Walker;
