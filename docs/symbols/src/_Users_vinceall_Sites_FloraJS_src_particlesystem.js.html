  2 /**
  3  * Creates a new ParticleSystem.
  4  *
  5  * @constructor
  6  * @extends Agent
  7  *
  8  * @param {Object} [opt_options] Particle options.
  9  * @param {boolean} [opt_options.isStatic = true] If set to true, particle system does not move.
 10  * @param {number} [opt_options.lifespan = -1] The number of frames before particle system dies. Set to -1 for infinite life.
 11  * @param {number} [opt_options.width = 0] Width
 12  * @param {number} [opt_options.height = 0] Height
 13  * @param {number} [opt_options.burst = 1] The number of particles to create per burst.
 14  * @param {number} [opt_options.burstRate = 1] The number of frames between bursts. Lower values = more particles.
 15  * @param {Object} [opt_options.particle = A particle at the system's location w random acceleration.] The particle to create. At minimum, should have a location vector. Use this.getLocation to get location of partilce system.
 16  */
 17  function ParticleSystem(opt_options) {
 18 
 19   'use strict';
 20 
 21   var options = opt_options || {},
 22       c = exports.defaultColors.getColor('heat'), // gets the heat start and end colors
 23       f = exports.defaultColors.getColor('food'), // gets the food start and end colors
 24       pl = new exports.ColorPalette();
 25 
 26   exports.Agent.call(this, options);
 27 
 28   pl.addColor({ // adds a random sampling of colors to palette
 29     min: 12,
 30     max: 24,
 31     startColor: c.startColor,
 32     endColor: c.endColor
 33   }).addColor({ // adds a random sampling of colors to palette
 34     min: 12,
 35     max: 24,
 36     startColor: f.startColor,
 37     endColor: f.endColor
 38   });
 39 
 40   this.beforeStep = function () {
 41 
 42     var i, max, p;
 43 
 44     if (this.world.clock % this.burstRate === 0) {
 45       for (i = 0; i < this.burst; i += 1) {
 46         p = new exports.Particle(this.particle());
 47       }
 48     }
 49     if (this.lifespan > 0) {
 50       this.lifespan -= 1;
 51     } else if (this.lifespan === 0) {
 52       exports.elementList.destroyElement(this.id);
 53     }
 54   };
 55   this.lifespan = options.lifespan || -1;
 56   this.width = options.width === 0 ? 0 : options.width || 0;
 57   this.height = options.height === 0 ? 0 : options.height || 0;
 58   this.isStatic = options.isStatic === false ? false : options.isStatic || true;
 59   this.burst = options.burst === 0 ? 0 : options.burst || 1;
 60   this.burstRate = options.burstRate === 0 ? 0 : options.burstRate || 3;
 61   this.particle = options.particle || function() {
 62 
 63     var borderStyles = exports.config.borderStyles,
 64         borderStr = borderStyles[exports.Utils.getRandomNumber(0, borderStyles.length - 1)];
 65 
 66     return {
 67       color: pl.getColor(),
 68       borderWidth: exports.Utils.getRandomNumber(2, 12),
 69       borderStyle: borderStr,
 70       borderColor: pl.getColor(),
 71       boxShadow: '0 0 0 ' + exports.Utils.getRandomNumber(2, 6) + 'px rgb(' + pl.getColor().toString() + ')',
 72       zIndex: exports.Utils.getRandomNumber(1, 100),
 73       location: this.getLocation(),
 74       acceleration: new exports.Vector(exports.Utils.getRandomNumber(-4, 4), exports.Utils.getRandomNumber(-4, 4))
 75     };
 76   };
 77 }
 78 exports.Utils.extend(ParticleSystem, exports.Agent);
 79 
 80 /**
 81  * Define a name property.
 82  */
 83 ParticleSystem.prototype.name = 'particlesystem';
 84 
 85 exports.ParticleSystem = ParticleSystem;
