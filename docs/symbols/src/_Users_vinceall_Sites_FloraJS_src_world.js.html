  2 /**
  3  * Creates a new World.
  4  *
  5  * @constructor
  6  *
  7  * @param {Object} [opt_options] World options.
  8  * @param {string} [opt_options.id = "m-" + World._idCount] An id. If an id is not provided, one is created.
  9  * @param {boolean} [opt_options.isStatic = true] Set to false if transforming the world every frame.
 10  * @param {number} [opt_options.clock = 0] Increments each frame.
 11  * @param {number} [opt_options.c = 0.01] Coefficient of friction.
 12  * @param {Object} [opt_options.gravity = {x: 0, y: 1}] Gravity
 13  * @param {Object} [opt_options.wind = {x: 0, y: 0}] Wind
 14  * @param {Object} [opt_options.location = {x: 0, y: 0}] Initial location
 15  * @param {number} [opt_options.scale = 1] Scale
 16  * @param {number} [opt_options.angle = 0] Angle
 17  * @param {number} [opt_options.opacity = 0.85] Opacity
 18  * @param {string} [opt_options.colorMode = 'rgb'] Color mode. Valid options are 'rgb'. 'hex' and 'hsl' coming soon.
 19  * @param {Array} [opt_options.color = null] The object's color expressed as an rbg or hsl value. ex: [255, 100, 0]
 20  * @param {number} [opt_options.borderWidth = 0] The border width.
 21  * @param {number} [opt_options.borderStyle = 0] The border style. Possible values: 'none', 'solid', 'dotted', 'dashed', 'double', 'inset', 'outset', 'groove', 'ridge'
 22  * @param {number} [opt_options.borderColor = null] The border color.
 23  * @param {number} [opt_options.borderRadius = 0] The border radius.
 24  * @param {number|string} [opt_options.boxShadow = 0] The box shadow.
 25  * @param {number} [opt_options.width = window.width] The world width.
 26  * @param {number} [opt_options.height = window.height] The world height.
 27  * @param {number} [opt_options.zIndex = 0] The world z-index.
 28  * @param {function} [opt_options.beforeStep = ''] A function to run before the step() function.
 29  * @param {function} [opt_options.afterStep = ''] A function to run after the step() function.
 30  */
 31 function World(opt_options) {
 32 
 33   'use strict';
 34 
 35   var me = this, options = opt_options || {},
 36       winSize = exports.Utils.getWindowSize();
 37 
 38   this.isStatic = options.isStatic || true;
 39   this.clock = options.clock || 0;
 40   this.c = options.c || 0.01;
 41   this.gravity = options.gravity || new exports.Vector(0, 1);
 42   this.wind =  options.wind || new exports.Vector();
 43   this.location = options.location || new exports.Vector();
 44 
 45   this.scale = options.scale || 1;
 46   this.angle = options.angle || 0;
 47   this.opacity = options.opacity || 1;
 48   this.colorMode = options.colorMode || 'rgb';
 49   this.color = options.color || [0, 0, 0];
 50   this.borderWidth = options.borderWidth || 0;
 51   this.borderStyle = options.borderStyle || 'none';
 52   this.borderColor = options.borderColor || null;
 53   this.borderRadius = options.borderRadius || 0;
 54   this.boxShadow = options.boxShadow || 0;
 55   this.zIndex = 0;
 56 
 57   this.beforeStep = options.beforeStep || undefined;
 58   this.afterStep = options.afterStep || undefined;
 59 
 60   /**
 61    * If no DOM element is passed for the world,
 62    * use document.body. Because the body initially has
 63    * no height, we use the window height.
 64    */
 65   if (!options.el) {
 66     this.el = document.body; // if no world element is passed, use document.body
 67     this.width = winSize.width;
 68     this.height = winSize.height;
 69     this.id = World.name + "-" + World._idCount; // if no id, create one
 70   } else {
 71     this.el = options.el;
 72     this.id = this.el.id; // use the element's id as the record id
 73     this.width = this.el.offsetWidth;
 74     this.height = this.el.offsetHeight;
 75   }
 76 
 77   this.el.className = 'world floraElement';
 78   this.el.style.width = this.width + 'px';
 79   this.el.style.height = this.height + 'px';
 80 
 81   World._idCount += 1; // increment id
 82 
 83   // events
 84 
 85   exports.Utils.addEvent(window, 'resize', function() { // listens for window resize
 86     me.resize.call(me);
 87   });
 88 }

 97 World._idCount = 0;

 99 /**
100  * Configures a new World by setting the DOM element and the element's width/height.
101  *
102  * @param {Object} opt_el The DOM element representing the world.
103  */
104 World.prototype.configure = function(opt_el) { // should be called after doc ready()
105 
106   'use strict';
107 
108   var el = opt_el || null;
109 
110   this.el = el || document.body; // if no world element is passed, use document.body
111   this.el.style.width = this.width + 'px';
112   this.el.style.height = this.height + 'px';
113 };

115 /**
116  * Updates a world instance with passed arguments.
117  * Typically called to change the window's default size or
118  * change the world's style.
119  *
120  * @param {Object} props A hash of properties to update.
121  *    Optional properties:
122  *    opt_props.style A map of css styles to apply.
123  */
124 World.prototype.update = function(opt_props) {
125 
126   'use strict';
127 
128   var i, key, cssText = '',
129       props = exports.Interface.getDataType(opt_props) === 'object' ? opt_props : {};
130 
131   for (key in props) {
132     if (props.hasOwnProperty(key)) {
133       this[key] = props[key];
134     }
135   }
136 
137   if (props.el) { // if updating the world's DOM element; update the width and height
138     this.width = parseInt(this.el.style.width.replace('px', ''), 10);
139     this.height = parseInt(this.el.style.height.replace('px', ''), 10);
140   }
141 
142   if (!this.el.style.setAttribute) { // WC3
143     if (props.style) {
144       for (i in props.style) {
145         if (props.style.hasOwnProperty(i)) {
146           this.el.style[i] = props.style[i];
147           cssText = cssText + i + ': ' + props.style[i] + ';';
148         }
149       }
150     }
151   } else { // IE
152     if (props.style) {
153       for (i in props.style) {
154         if (props.style.hasOwnProperty(i)) {
155           cssText = cssText + i + ': ' + props.style[i] + ';';
156         }
157       }
158     }
159     this.el.style.setAttribute('cssText', cssText, 0);
160   }
161 };

163 /**
164  * Called from a window resize event, resize() repositions all Flora elements relative
165  * to the new window size. Also, if the world is the document.body, resets the body's
166  * width and height attributes.
167  */
168 World.prototype.resize = function() {
169 
170   'use strict';
171 
172   var i, max, elementLoc, controlCamera, winSize = exports.Utils.getWindowSize(),
173     windowWidth = winSize.width,
174     windowHeight = winSize.height,
175     elements = exports.elementList.all();
176 
177   // check of any elements control the camera
178   for (i = 0, max = elements.length; i < max; i += 1) {
179     if (elements[i].controlCamera) {
180       controlCamera = true;
181       break;
182     }
183   }
184 
185   // loop thru elements
186   if (!controlCamera) {
187     for (i = 0, max = elements.length; i < max; i += 1) {
188 
189       elementLoc = elements[i].location; // recalculate location
190 
191       elementLoc.x = windowWidth * (elementLoc.x/this.width);
192       elementLoc.y = windowHeight * (elementLoc.y/this.height);
193 
194     }
195 
196     if (this.el === document.body) {
197       this.width = windowWidth;
198       this.height = windowHeight;
199       this.el.style.width = this.width + 'px';
200       this.el.style.height = this.height + 'px';
201     }
202   }
203 };

205 /**
206  * Called every frame, step() updates the world's properties.
207  */
208 World.prototype.step = function() {
209 
210   'use strict';
211 
212   if (this.beforeStep) {
213     this.beforeStep.apply(this);
214   }
215   if (this.afterStep) {
216     this.afterStep.apply(this);
217   }
218 };
219 
220 /**
221  * Called every frame, draw() renders the world.
222  */
223 World.prototype.draw = function() {
224 
225   'use strict';
226 
227   /**
228    * If there's not an object controlling the camera,
229    * we want to draw the world once.
230    */
231   if (!exports.camera.controlObj && this.isStatic && this.clock > 0) {
232     return;
233   }
234 
235   this.el.style.cssText = exports.Utils.getCSSText({
236     x: this.location.x,
237     y: this.location.y,
238     s: this.scale,
239     a: this.angle,
240     o: this.opacity,
241     w: this.width,
242     h: this.height,
243     cm: this.colorMode,
244     color: this.color,
245     z: this.zIndex,
246     borderWidth: this.borderWidth,
247     borderStyle: this.borderStyle,
248     borderColor: this.borderColor,
249     borderRadius: this.borderRadius,
250     boxShadow: this.boxShadow
251   });
252 };
253 exports.World = World;
