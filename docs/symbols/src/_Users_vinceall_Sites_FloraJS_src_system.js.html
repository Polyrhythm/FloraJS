1 /*global exports, window */
  2 /**
  3  * Creates a new System.
  4  *
  5  * @constructor
  6  */
  7 function System() {
  8 
  9   'use strict';
 10 }
 11 
 12 /**
 13  * Define a name property.
 14  */
 15 System.name = 'system';
 16 
 17 /**
 18  * A list of instructions to execute before the system starts.
 19  */
 20 System.setup = null;
 21 
 22 /**
 23  * Starts the System.
 24  * @param {function} func A list of instructions to execute before the system starts.
 25  * @param {Object} opt_universe A map of options for the System's Universe.
 26  * @param {Array} opt_worlds An array of DOM elements to use as Worlds.
 27  */
 28 System.start = function (func, opt_universe, opt_worlds) {
 29 
 30   'use strict';
 31 
 32   var i, max,
 33       defaultColorList = exports.config.defaultColorList;
 34 
 35   this.universeOptions = opt_universe || null;
 36   this.worlds = opt_worlds || null;
 37 
 38   this._featureDetector = new exports.FeatureDetector();
 39 
 40   this.supportedFeatures = {};
 41   this.supportedFeatures.csstransforms = this._featureDetector.detect('csstransforms');
 42   this.supportedFeatures.csstransforms3d = this._featureDetector.detect('csstransforms3d');
 43   this.supportedFeatures.touch = this._featureDetector.detect('touch');
 44 
 45   exports.liquids = [];
 46   exports.repellers = [];
 47   exports.attractors = [];
 48   exports.heats = [];
 49   exports.colds = [];
 50   exports.lights = [];
 51   exports.oxygen = [];
 52   exports.food = [];
 53   exports.predators = [];
 54 
 55   exports.mouse = {
 56     loc: new exports.Vector(),
 57     locLast: new exports.Vector()
 58   };
 59 
 60   // create elementList before universe
 61   exports.elementList = new exports.ElementList();
 62 
 63   exports.universe = new exports.Universe(this.universeOptions);
 64   if (exports.Interface.getDataType(this.worlds) === 'array') {
 65     for (i = 0, max = this.worlds.length; i < max; i += 1) {
 66       exports.universe.addWorld({
 67         el: this.worlds[i]
 68       });
 69     }
 70   } else {
 71     exports.universe.addWorld();
 72   }
 73 
 74   exports.camera = new exports.Camera();
 75 
 76   // add default colors
 77   exports.defaultColors = new exports.ColorTable();
 78   for (i = 0, max = defaultColorList.length; i < max; i++) {
 79     exports.defaultColors.addColor({
 80       name: defaultColorList[i].name,
 81       startColor: defaultColorList[i].startColor,
 82       endColor: defaultColorList[i].endColor
 83     });
 84   }
 85 
 86   exports.animLoop = function () {
 87 
 88     var i, max,
 89         universe = exports.universe,
 90         world = universe.first(),
 91         elements = exports.elementList.all();
 92 
 93     if (universe.isPlaying) {
 94       window.requestAnimFrame(exports.animLoop);
 95 
 96       if (universe.zSorted) {
 97         elements = elements.sort(function(a,b){return (b.zIndex - a.zIndex);});
 98       }
 99 
100       for (i = elements.length - 1; i >= 0; i -= 1) {
101         elements[i].step();
102         if (elements[i]) {
103           elements[i].draw();
104         }
105       }
106 
107       exports.universe.updateClocks();
108     }
109   };
110 
111 
112   func = exports.Interface.getDataType(func) === "function" ? func : function () {};
113   System.setup = func;
114 
115   func();
116   exports.animLoop();
117 };
118 
119 /**
120  * Destroys a System.
121  */
122 System.destroy = function () {
123   'use strict';
124   exports.elementList.destroyAll();
125 };
126 
127 exports.System = System;
