1 /*global exports, window */
  2 /**
  3  * Creates a new System.
  4  *
  5  * @constructor
  6  */
  7 function System(opt_options) {
  8 
  9   'use strict';
 10 
 11   var i, max,
 12       options = opt_options || {},
 13       defaultColorList = exports.config.defaultColorList;
 14 
 15   this.el = options.world || null;
 16   this.universeOptions = options.universeOptions || null;
 17 
 18   exports.liquids = [];
 19   exports.repellers = [];
 20   exports.attractors = [];
 21   exports.heats = [];
 22   exports.colds = [];
 23   exports.lights = [];
 24   exports.oxygen = [];
 25   exports.food = [];
 26   exports.predators = [];
 27 
 28   exports.mouse = {
 29     loc: new exports.Vector(),
 30     locLast: new exports.Vector()
 31   };
 32 
 33   // create elementList before universe
 34   exports.elementList = new exports.ElementList();
 35 
 36   exports.universe = new exports.Universe(this.universeOptions);
 37   if (exports.Interface.getDataType(this.el) === 'array') {
 38     for (i = 0, max = this.el.length; i < max; i += 1) {
 39       exports.universe.addWorld({
 40         el: this.el[i]
 41       });
 42     }
 43   } else {
 44     exports.universe.addWorld({
 45       el: this.el
 46     });
 47   }
 48 
 49   exports.camera = new exports.Camera();
 50 
 51   // add default colors
 52   exports.defaultColors = new exports.ColorTable();
 53   for (i = 0, max = defaultColorList.length; i < max; i++) {
 54     exports.defaultColors.addColor({
 55       name: defaultColorList[i].name,
 56       startColor: defaultColorList[i].startColor,
 57       endColor: defaultColorList[i].endColor
 58     });
 59   }
 60 
 61   exports.animLoop = function () {
 62 
 63     var i, max,
 64         universe = exports.universe,
 65         world = universe.first(),
 66         elements = exports.elementList.all();
 67 
 68     if (universe.isPlaying) {
 69       window.requestAnimFrame(exports.animLoop);
 70 
 71       if (universe.zSorted) {
 72         elements = elements.sort(function(a,b){return (b.zIndex - a.zIndex);});
 73       }
 74 
 75       for (i = elements.length - 1; i >= 0; i -= 1) {
 76         elements[i].step();
 77         if (elements[i]) {
 78           elements[i].draw();
 79         }
 80       }
 81 
 82       exports.universe.updateClocks();
 83     }
 84   };
 85 }
 86 
 87 /**
 88  * Define a name property.
 89  */
 90 System.prototype.name = 'system';
 91 
 92 /**
 93  * A list of instructions to execute before the system starts.
 94  */
 95 System.setup = null;
 96 
 97 /**
 98  * Starts a System.
 99  * @param {function} func A list of instructions to execute before the system starts.
100  */
101 System.prototype.start = function (func) {
102 
103   'use strict';
104 
105   func = exports.Interface.getDataType(func) === "function" ? func : function () {};
106   System.setup = func;
107 
108   func();
109   exports.animLoop();
110 };
111 
112 System.prototype.destroy = function () {
113   'use strict';
114   exports.elementList.destroyAll();
115 };
116 
117 exports.System = System;
